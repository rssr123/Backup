create procedure sp_updatelatestorderstatus(i_mtt_id int)
returning int as result;

define row_no int;
define status_found int;
define latest_status nvarchar(10);
define o_result int;

-- check if mtt exists
select count(*) into row_no
from rms_mtt_pg
where mtt_id = i_mtt_id;

    if row_no > 0 then

        -- check if there is a payment record with pg_txn_status = 0 for the mtt_id
        select count(*) 
        into status_found
        from rms_mtt_pg
        where mtt_id = i_mtt_id
        and pg_txn_status = 0;

        if status_found > 0 then
            -- if a record with pg_txn_status = 0 is found, return 0
            let o_result = 0;
        else
            -- if no record with pg_txn_status = 0 is found, find the latest pg_txn_status
            foreach c_something for

                select first 1 pg_txn_status 
                into o_result
                from rms_mtt_pg
                where mtt_id = i_mtt_id
                order by dt_created desc
            end foreach;
            
        end if;

        -- lookup proper name for order status
        case o_result
            when 0 then let latest_status = 'p'; -- "paid"
            when 1 then let latest_status = 'f'; -- "failed";
            when 2 then let latest_status = 'pp'; -- "pending payment"
            else let latest_status = 'pp';
        end case;

        -- update mtt's order status
        update rms_mtt
        set order_status = latest_status
        where mtt_id = i_mtt_id;

    end if;

return o_result;

end procedure;

create procedure sp_upddi()
returning int as result;

define o_result int;

on exception
	rollback work;
end exception;

begin work;
--set lock mode to wait 10;

select count(1)
into o_result
from rmd_di
where status not in ('C','T')
and next_calc_dt <= today;

if o_result > 0 then

    -- create a temporary table
    create temp table tempdi (
        di_id int8,
        fee_detail_pk int,
        fee_detail_id nvarchar(10),
        entity_type nvarchar(1),
        entity_no nvarchar(20),
        unit_fee decimal(16,2),
        bal_di_amt decimal(16,2),
        next_calc_dt date,
        status nvarchar(10)
    );

    -- complete terminated exclude
    insert into tempdi (di_id, unit_fee, bal_di_amt, next_calc_dt, status)
    select di_id, unit_fee, bal_di_amt, next_calc_dt, status
    from rms_di
    where status not in ('C','T')
    and next_calc_dt <= today;

    -- insert statement rms_di audit
    insert into rms_di_a(di_id, fee_detail_pk, fee_detail_id, dt_txn, action_type, 
                            entity_type, entity_no, bal_di_amt_b4, bal_di_amt_af, status_b4, status_af, 
                                dt_created, dt_modified, created_by, modified_by, status) 
    select di_id, fee_detail_pk, fee_detail_id, current, upper('pca'),
            entity_type, entity_no, bal_di_amt, bal_di_amt - unit_fee, status, 
            case 
                when bal_di_amt - unit_fee > 0 
                    then upper('a')
                else upper('c')
            end,
                current, current, 'system', 'system', upper('a')
    from rms_di 
    where status not in ('C','T')
    and next_calc_dt <= today;

    -- update statement rms_di
    update rms_di
    set t1.bal_di_amt = t2.bal_di_amt - t2.unit_fee,
        t1.next_calc_dt = extend(extend(t2.next_calc_dt, year to day) + 1 units year, year to day) - 1 units day,
        t1.dt_modified = current,
        t1.status = case 
                    when t2.bal_di_amt - t2.unit_fee > 0 
                        then upper('a')
                    else upper('c')
                 end
    from rms_di t1
    inner join tempdi t2 on t1.di_id = t2.di_id;

    let o_result = dbinfo('sqlca.sqlerrd2');

    -- insert statement rms_di journal debit

    -- insert statement rms_di journal credit

    drop table tempdi;

else

    let o_result = 0;

end if;

commit work;

return o_result;

end procedure

CREATE PROCEDURE sp_updfms(
    i_fms_id INT, 
    i_fms_cd NVARCHAR(50), 
    i_modified_by NVARCHAR(25),  
    i_is_active INT
)
RETURNING INT AS row_count;

DEFINE o_row_count INT;
DEFINE is_other_active INT;

IF i_is_active = 0 THEN
    -- Check if there is an FMS currently active
    SELECT COUNT(*)
    INTO is_other_active
    FROM rms_fms
    WHERE is_active = 1 and fms_id <> i_fms_id;

    -- If the FMS is currently active, deactivate it
    IF is_other_active > 0 THEN
        UPDATE rms_fms
        SET is_active = 0
        WHERE fms_id = i_fms_id;
    END IF;
END IF;


-- Check if i_is_active is 1
IF i_is_active = 1 THEN
    -- Check if other FMS records are active
    SELECT COUNT(*)
    INTO is_other_active
    FROM rms_fms
    WHERE is_active = 1 AND fms_id <> i_fms_id;

    -- If other FMS records are active, set them to not active
    IF is_other_active > 0 THEN
        UPDATE rms_fms
        SET is_active = 0
        WHERE is_active = 1 AND fms_id <> i_fms_id;
    END IF;

     UPDATE rms_fms
        SET is_active = 1
        WHERE is_active = 0 AND fms_id = i_fms_id;
END IF;

-- Update the specified FMS record
UPDATE rms_fms
SET
    fms_cd = CASE WHEN i_fms_cd IS NULL THEN fms_cd ELSE i_fms_cd END,
    dt_modified = CURRENT,
    modified_by = i_modified_by
WHERE fms_id = i_fms_id;

-- Get the row count affected by the update
LET o_row_count = DBINFO('sqlca.sqlerrd2');

RETURN o_row_count;

END PROCEDURE;

CREATE PROCEDURE sp_updpgmtttxn()
returning int8 as result;
 
define o_result int;
define o_exist int;
define o_compare_range int;
define o_dt_settlement date;

on exception
	rollback work;
end exception;
 
-- Set a default value
LET o_result = '0';
 
begin work;
set lock mode to wait 30;

-- create a temporary table
create temp table temppgmtt (
    rc_pgmtt_id int8,
    rc_pg_id int8,
    sub_criteria nvarchar(10)
);
 
-- check exist
select count(1)
into o_exist
from rms_rc_pgmtt
where sub_criteria = 'NF';

if o_exist > 0 then

    -- select statement
    insert into temppgmtt (rc_pgmtt_id,rc_pg_id,sub_criteria)
    SELECT 
        rms_rc_pgmtt.rc_pgmtt_id, 
        rms_rc_pgmtt.rc_pg_id,
        CASE 
            WHEN rms_rc_pgtxn.txn_id IS NULL THEN 'NFP'
            WHEN rms_rc_pg.task_status = 'CL' THEN 'CIP'
            ELSE 'NCP'
        END as Status
    FROM 
        rms_rc_pgmtt
    LEFT JOIN 
        rms_mtt_pg ON rms_rc_pgmtt.mtt_pg_id = rms_mtt_pg.mtt_pg_id
    LEFT JOIN 
        rms_rc_pgtxn ON rms_mtt_pg.pg_txn_id = rms_rc_pgtxn.txn_id
    LEFT JOIN 
        rms_rc_pg ON rms_rc_pgtxn.rc_pg_id = rms_rc_pg.rc_pg_id
    WHERE 
        rms_rc_pgmtt.sub_criteria = 'NF';

    -- update
    UPDATE 
        rms_rc_pgmtt
    SET 
        rms_rc_pgmtt.sub_criteria = temppgmtt.sub_criteria
    FROM 
        temppgmtt
    WHERE 
        rms_rc_pgmtt.rc_pgmtt_id = temppgmtt.rc_pgmtt_id;
 
    -- get update count
    let o_result = dbinfo('sqlca.sqlerrd2');

    -- update rms_rc_pg' recon status
    update rms_rc_pg
    set recon_status = 'PSSCR'
    where rc_pg_id in (select distinct rc_pg_id from temppgmtt);

drop table temppgmtt;
 
end if;
 
commit work;
 
if o_result > 0 then
    return o_result;
else
    return -1;
end if;
 
end procedure
GO

	rollback work;

create procedure sp_updrcbanktxn()
returning int as result;

define o_txn_ref nvarchar(50);
define o_rc_bank_id int8;
define o_total_credit decimal(15, 2);
define o_stmt_no nvarchar(25);
define o_total_paid decimal(15, 2);
define o_total_count int;
define o_status int;

on exception
	rollback work;
end exception;

begin work;

let o_status = 1;

select count(txn_ref)
into o_total_count
from rms_rc_banktxn
where status = 'PBSR';

while o_total_count > 0

    -- Select txn_ref
    select first 1 txn_ref
    into o_txn_ref
    from rms_rc_banktxn
    where status = 'PBSR';

    -- Select from rms_rc_banktxn table
    select rc_bank_id, sum(credit)
    into o_rc_bank_id, o_total_credit
    from rms_rc_banktxn
    where status = 'PBSR'
    and txn_ref like substring(o_txn_ref from 0 for 15) || '%'
    group by rc_bank_id;

    -- Select from pg table
    select total_paid
    into o_total_paid
    from rms_rc_pg
    where o_txn_ref like '%' || stmt_no || '%';

    if(o_total_paid <> o_total_credit) then
            -- Update status in bank txn 
            update rms_rc_banktxn
            set status = 'FBSR',
                dt_modified = current
            where txn_ref like substring(o_txn_ref from 0 for 15) || '%';

            -- Update status in rc pg
            update rms_rc_pg
            set recon_status = 'FBSR',
                dt_modified = current
            where o_txn_ref like '%' || stmt_no || '%'; 

            -- Update status in rc bank
            update rms_rc_bank
            set recon_status = 'FBSR',
                dt_modified = current
            where rc_bank_id = o_rc_bank_id;

            -- Set status = failed
            let o_status = 0;
        else
            -- Update status in bank txn 
            update rms_rc_banktxn
            set status = 'SBSR',
                dt_modified = current
            where txn_ref like substring(o_txn_ref from 0 for 15) || '%';

            -- Update status in rc pg
            update rms_rc_pg
            set recon_status = 'SBSR',
                dt_modified = current
            where o_txn_ref like '%' || stmt_no || '%'; 

            -- Update status in rc pg
            update rms_rc_bank
            set recon_status = 'SBSR',
                dt_modified = current
            where rc_bank_id = o_rc_bank_id;

            -- Set status = success
            let o_status = 1;
        end if;

let o_total_count = o_total_count - 1;

end while;

commit work;

return o_status;

end procedure;

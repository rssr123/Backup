create procedure sp_getbanknostmt(i_page integer,i_size integer, i_task_no nvarchar(25), i_file_nm nvarchar(255), i_uploaded_by nvarchar(255), i_dt_uploaded date)
returning nvarchar(255) as file_nm, integer as file_size_kb, nvarchar(255) as uploaded_by, date as dt_uploaded, integer as total;

define o_file_nm nvarchar(255);
define o_file_size integer;
define o_uploaded_by nvarchar(255);
define o_dt_uploaded date;
define calculatedpage integer;
define total integer;

if (i_page is null or i_page < 1) or (i_size is null or i_page < 1) then 
    raise exception -746, 0, "i_page or i_size is null integer";
end if

    let calculatedpage = (i_page-1)*i_size;
    let total = (select count(1) 
                from rms_rc_bankdoc t1
                left join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
                left join rms_user t3 on t1.created_by = t3.ssm4uuserrefno
                where (i_task_no is null or t2.task_no = i_task_no)
                and (i_file_nm is null or t1.file_nm like '%' || i_file_nm || '%')
                and (i_uploaded_by is null or t3.nm = i_uploaded_by)
                and (i_dt_uploaded is null or t1.dt_created = i_dt_uploaded));

    foreach c_sp_getbanknostmt for
		select skip calculatedpage limit i_size t1.file_nm, t1.file_size_kb , t3.nm, t1.dt_created
		into o_file_nm, o_file_size, o_uploaded_by, o_dt_uploaded
	    from rms_rc_bankdoc t1
        left join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
        left join rms_user t3 on t1.created_by = t3.ssm4uuserrefno
	    where (i_task_no is null or t2.task_no = i_task_no)
                and (i_file_nm is null or t1.file_nm like '%' || i_file_nm || '%')
                and (i_uploaded_by is null or t3.nm = i_uploaded_by)
                and (i_dt_uploaded is null or t1.dt_created = i_dt_uploaded)

		return o_file_nm, o_file_size, o_uploaded_by, o_dt_uploaded, total with resume;
	end foreach

end procedure;
GO
create procedure sp_getbankpgfiletxn(i_page integer,i_size integer, i_task_no nvarchar(25), i_txn_ref nvarchar(50), i_dt_posting date)
returning nvarchar(50) as txn_ref, decimal(15,5) as credit, date as dt_posting, integer as total;

define o_txn_ref nvarchar(50);
define o_credit decimal(15,5);
define o_dt_posting date;
define calculatedpage integer;
define total integer;

if (i_page is null or i_page < 1) or (i_size is null or i_page < 1) then 
    raise exception -746, 0, "i_page or i_size is null integer";
end if

    let calculatedpage = (i_page-1)*i_size;
    let total = (select count(1)
                 from(
                    select t1.txn_ref, t1.dt_posting, t1.credit
                    from rms_rc_banktxn t1
                    inner join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
                    where t2.task_no = i_task_no
                    and substring(t1.txn_ref from 0 for 14) in (select stmt_no 
                                                                from rms_rc_pg t1
                                                                inner join rms_rc_bank t2 on t1.rc_pg_id = t2.rc_pg_id)
                    )t3
                 where (i_txn_ref is null or t3.txn_ref = i_txn_ref)
                 and  (i_dt_posting is null or t3.dt_posting = i_dt_posting)
                 );

    foreach c_getbankpgfiletxn for
		select skip calculatedpage limit i_size t3.txn_ref, t3.credit, t3.dt_posting
        into o_txn_ref, o_credit, o_dt_posting
        from(
             select t1.txn_ref, t1.dt_posting, t1.credit
             from rms_rc_banktxn t1
             inner join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
             where t2.task_no = i_task_no
             and substring(t1.txn_ref from 0 for 14) in (select stmt_no 
                                                         from rms_rc_pg t1
                                                         inner join rms_rc_bank t2 on t1.rc_pg_id = t2.rc_pg_id)
             )t3
         where (i_txn_ref is null or t3.txn_ref = i_txn_ref)
         and  (i_dt_posting is null or t3.dt_posting = i_dt_posting)

		return o_txn_ref, o_credit, o_dt_posting, total with resume;
	end foreach

end procedure;
GO
create procedure sp_getbankpgtxnlisting(i_page integer, i_size integer, i_task_no nvarchar(25), i_txn_date date,
                                    i_txn_type nvarchar(25), i_fnd_pg nvarchar(3), i_txn_id nvarchar(30), i_txn_cd nvarchar(50))
returning DateTime year to second as dt_txn,
          nvarchar(30) as txn_id,
          nvarchar(25) as txn_type,
          nvarchar(25) as txn_cd,
          nvarchar(3) as found_in_pg,
          --nvarchar(25) as sub_criteria,
          decimal(15,2) as txn_amt,
          decimal(15,2) as mdr_amt,
          decimal(15,2) as sst_amt,
          decimal(15,2) as net_amt,
          int as total;

define o_txn_date           DateTime year to second;
define o_txn_id             nvarchar(30);
define o_txn_type           nvarchar(25);
define o_txn_cd             nvarchar(25);
define c_fnd_in_pg          int;
define o_fnd_in_pg          nvarchar(3);
--define o_sub_cri            nvarchar(25);
define o_txn_amt            decimal(15,2);
define o_mdr_amt            decimal(15,2);
define o_sst_amt            decimal(15,2);
define o_net_amt            decimal(15,2);
define calculatedpage		int;
define total	 			int;

if (i_page is null or i_page < 1) or (i_size is null or i_page < 1) then 
    raise exception -746, 0, "i_page or i_size is null integer";
end if;

--temp set to null
let i_txn_date = null;

if(i_fnd_pg is not null and i_fnd_pg = 'Yes') then
    let c_fnd_in_pg = 1;
elif (i_fnd_pg is not null and i_fnd_pg = 'No') then
    let c_fnd_in_pg = 0;
else
    let c_fnd_in_pg = null;
end if;

let calculatedpage = (i_page-1)*i_size;
let total = (select count(1)
             from rms_rc_bank t1
             inner join rms_rc_pgtxn t2 on t1.rc_pg_id = t2.rc_pg_id
             where (i_task_no is null or t1.task_no = i_task_no)
             and (i_txn_date is null or t2.dt_txn = i_txn_date)
             and (i_txn_type is null or t2.txn_type like '%' || i_txn_type || '%')
             and (c_fnd_in_pg is null or t2.found_in_rms = c_fnd_in_pg)
             and (i_txn_id is null or t2.txn_id like '%' || i_txn_id || '%')
             and (i_txn_cd is null or t2.txn_cd like '%' || i_txn_cd || '%')
             --and (i_sub_cri is null or t2.sub_criteria like '%' || i_sub_cri || '%')
             );

foreach c_getbankpgtxnlisting for
    select skip calculatedpage limit i_size t2.dt_txn, t2.txn_id, t2.txn_type, t2.txn_cd, 
            case when t2.found_in_rms = 1 then 'Yes'
                else 'No'
            end found_in_rms, 
           --t2.sub_criteria,
           t2.txn_amt, t2.mdr_amt, t2.sst_amt, t2.net_amt
    into o_txn_date, o_txn_id, o_txn_type, o_txn_cd, o_fnd_in_pg, 
         --o_sub_cri, 
         o_txn_amt, o_mdr_amt, o_sst_amt, o_net_amt 
    from rms_rc_bank t1
    inner join rms_rc_pgtxn t2 on t1.rc_pg_id = t2.rc_pg_id
    where (i_task_no is null or t1.task_no = i_task_no)
    and (i_txn_date is null or t2.dt_txn = i_txn_date)
	and (i_txn_type is null or t2.txn_type like '%' || i_txn_type || '%')
    and (c_fnd_in_pg is null or t2.found_in_rms = c_fnd_in_pg)
    and (i_txn_id is null or t2.txn_id like '%' || i_txn_id || '%')
    and (i_txn_cd is null or t2.txn_cd like '%' || i_txn_cd || '%')
    --and (i_sub_cri is null or t2.sub_criteria like '%' || i_sub_cri || '%')

    return o_txn_date, o_txn_id, o_txn_type, o_txn_cd, o_fnd_in_pg, 
           --o_sub_cri,
           o_txn_amt, o_mdr_amt, o_sst_amt, o_net_amt, total with resume;

end foreach;

end procedure

create procedure sp_getBankRecon(i_page integer, i_size integer, i_rc_bank_id int8, i_task_no nvarchar(25), i_dt_settlement nvarchar(100), 
								i_merchant_id nvarchar(25), i_task_status nvarchar(25), i_dt_upload nvarchar(100), 
								i_recon_status nvarchar(25))
								
returning 
		int8 as o_rc_bank_id,
		nvarchar(40) as o_task_no,
		nvarchar(100) as o_dt_settlement,
		nvarchar(50) as o_merchant_id,
		nvarchar(50) as o_task_status,
		nvarchar(100) as o_dt_upload,
		nvarchar(50) as o_recon_status,
		integer as total;
	
	define total integer;
	define calculatedpage integer;
	define o_rc_bank_id int8;
	define o_task_no nvarchar(40);
	define o_dt_settlement nvarchar(100);
	define o_merchant_id nvarchar(50);
	define o_task_status nvarchar(50);
	define o_dt_upload nvarchar(100);
	define o_recon_status nvarchar(50);
	
if (i_page is null or i_page < 1) or (i_size is null or i_page < 1) then 
        raise exception -746, 0, "i_page or i_size is null integer";
    end if

	let calculatedpage = (i_page-1)*i_size;
	let total = (select count(1) 
		from rms_rc_bank T1
        left join rms_param T2 on T1.task_status = T2.param_cd and T2.param_grp_nm='Status-Task'
        left join rms_param T3 on T1.recon_status= T3.param_cd and T3.param_grp_nm='Status-BRC'
        left join rms_rc_pg T4 on T1.rc_pg_id=T4.rc_pg_id 
--        left join rms_param T4 on T1.txn_type= T4.param_cd and T4.param_grp_nm='RIPL-TxnType'
        where (i_rc_bank_id is null or rc_bank_id = i_rc_bank_id)
        				  and (i_task_no is null or T1.task_no like '%' || i_task_no || '%')
                          and (i_dt_settlement is null or date(T4.dt_settlement) == to_date(i_dt_settlement,'%Y-%m-%d')::Date)
                          and (i_merchant_id is null or T4.merchant_id like '%' || i_merchant_id || '%')
                          and (i_task_status is null or T1.task_status == i_task_status)
                          and (i_dt_upload is null or date(T1.dt_upload) == to_date(i_dt_upload,'%Y-%m-%d')::Date)
                          and (i_recon_status is null or T1.recon_status == i_recon_status)
        			);

	foreach c_getBankReconTask for
        select  skip calculatedpage limit i_size T1.rc_bank_id, T1.task_no, TO_CHAR(T4.dt_settlement, '%d %b %Y'),
        T4.merchant_id, T2.nm_en, TO_CHAR(T1.dt_upload, '%d %b %Y %H:%M:%S'), T3.nm_en
        into o_rc_bank_id,o_task_no, o_dt_settlement, o_merchant_id, o_task_status, o_dt_upload, o_recon_status
        from rms_rc_bank T1
        left join rms_param T2 on T1.task_status = T2.param_cd and T2.param_grp_nm='Status-Task'
        left join rms_param T3 on T1.recon_status= T3.param_cd and T3.param_grp_nm='Status-BRC'
        left join rms_rc_pg T4 on T1.rc_pg_id=T4.rc_pg_id 
--        left join rms_param T4 on T1.txn_type= T4.param_cd and T4.param_grp_nm='RIPL-TxnType'
        where (i_rc_bank_id is null or rc_bank_id = i_rc_bank_id)
        				  and (i_task_no is null or T1.task_no like '%' || i_task_no || '%')
                          and (i_dt_settlement is null or date(T4.dt_settlement) == to_date(i_dt_settlement,'%Y-%m-%d')::Date)
                          and (i_merchant_id is null or T4.merchant_id like '%' || i_merchant_id || '%')
                          and (i_task_status is null or T1.task_status == i_task_status)
                          and (i_dt_upload is null or date(T1.dt_upload) == to_date(i_dt_upload,'%Y-%m-%d')::Date)
                          and (i_recon_status is null or T1.recon_status == i_recon_status)
        order by T4.dt_settlement desc
	
		return o_rc_bank_id,o_task_no, o_dt_settlement, o_merchant_id, o_task_status, o_dt_upload, o_recon_status, total with resume;
	end foreach

end procedure

CREATE PROCEDURE sp_getconfig(i_config_cd NVARCHAR(100))
RETURNING NVARCHAR(255) AS config_value;

DEFINE o_config_value NVARCHAR(255);

	FOREACH c_getconfig FOR
		SELECT config_value
		INTO o_config_value
		FROM rms_config
		WHERE (i_config_cd IS NULL OR config_cd = i_config_cd)
		
		RETURN o_config_value WITH RESUME;
	END FOREACH

LET o_mft_total = (SELECT COUNT(1) FROM rms_mft where status = 'A');

foreach c_getfms for
    select skip calculatedpage limit i_size t1.fms_id, t1.fms_cd,  count(case when t2.found = 'F' then 1 else null end), 
        t1.modified_by, t1.is_active, t1.dt_modified
    into o_fms_id, o_fms_cd, o_cong_led_cnt, o_modified_by, o_is_active, o_dt_modified
    from rms_fms t1
    left outer join rms_fms_ledger t2 on t1.fms_id = t2.fms_id
    left outer join rms_user t3 on t1.modified_by = t3.ssm4uuserrefno
    where (i_fms_cd is null or t1.fms_cd like '%' || i_fms_cd || '%')
        and (i_modified_by is null or i_modified_by like '%' || t3.nm || '%')
        and (i_dt_modified_fr is null or t1.dt_modified >= i_dt_modified_fr)
        and (i_dt_modified_to is null or t1.dt_modified <= i_dt_modified_to)
        and (i_status is null or  t1.status = i_status)
    group by t1.fms_id, t1.fms_cd, t1.modified_by, t3.nm, t1.is_active, t1.dt_modified

CREATE PROCEDURE sp_getfmsledger_v2(
    i_page INTEGER,
    i_size INTEGER,
    i_fms_id INTEGER,
    i_file_index INTEGER,
    i_fee_detail_id NVARCHAR (10),
    i_fee_detail_nm_en NVARCHAR (100),
    i_fms_ledger_cd NVARCHAR(50),
    i_found NVARCHAR(2)
)
RETURNING
    NVARCHAR(50) AS fms_ledger_cd,
    NVARCHAR(10) AS fee_detail_id,
    NVARCHAR(100) AS fee_detail_nm_en,
    NVARCHAR(2) AS found,
    NVARCHAR(25) AS fms_cd,
    INTEGER AS mft_total,
    INTEGER AS total;

DEFINE o_fms_ledger_cd NVARCHAR(50);
DEFINE o_fee_detail_id NVARCHAR(10);
DEFINE o_fee_detail_nm_en NVARCHAR(100);
DEFINE o_found NVARCHAR(2);
DEFINE o_fms_cd NVARCHAR(25);
DEFINE total INTEGER;
DEFINE calculatedpage INT;
DEFINE o_mft_total INTEGER;

IF (i_page IS NULL OR i_page < 1) OR (i_size IS NULL OR i_size < 1) THEN
    RAISE EXCEPTION -746, 0, "i_page or i_size is null or less than 1";
END IF;

-- Calculate total count
LET total = (SELECT COUNT(1)
             FROM rms_fms_ledger t1
             WHERE (i_fms_id IS NULL OR t1.fms_id = i_fms_id)
               AND (i_file_index IS NULL OR t1.file_index = i_file_index)
               AND (i_fee_detail_id IS NULL OR t1.fee_detail_id LIKE '%' || i_fee_detail_id || '%')
               AND (i_fee_detail_nm_en IS NULL OR t1.fee_detail_nm_en LIKE '%' || i_fee_detail_nm_en || '%')
               AND (i_fms_ledger_cd IS NULL OR t1.fms_ledger_cd LIKE '%' || i_fms_ledger_cd || '%')
               AND (i_found IS NULL OR i_found = t1.found)
                AND (t1.status = 'A'));

LET calculatedpage = (i_page - 1) * i_size;

LET o_mft_total = (SELECT COUNT(1) FROM rms_mft);

FOREACH c_getfmsledger FOR
        SELECT SKIP calculatedpage LIMIT i_size
            t1.fms_ledger_cd, 
            t1.fee_detail_id, 
            t1.fee_detail_nm_en,
            t3.fms_cd,
            t1.found
        INTO o_fms_ledger_cd, 
             o_fee_detail_id, 
             o_fee_detail_nm_en,
             o_fms_cd,
             o_found
FROM rms_fms_ledger t1
         LEFT OUTER JOIN rms_fms t3 ON t3.fms_id = t1.fms_id
         LEFT OUTER JOIN rms_fms_doc t2 ON t2.file_index = t1.file_index
         LEFT OUTER JOIN rms_mft t4 ON t4.fee_detail_id = t1.fee_detail_id
WHERE (i_fms_id IS NULL OR t1.fms_id = i_fms_id)
  AND (i_file_index IS NULL OR t1.file_index = i_file_index)
  AND (i_fee_detail_id IS NULL OR t1.fee_detail_id LIKE '%' || i_fee_detail_id || '%')
  AND (i_fee_detail_nm_en IS NULL OR t1.fee_detail_nm_en LIKE '%' || i_fee_detail_nm_en || '%')
  AND (i_fms_ledger_cd IS NULL OR t1.fms_ledger_cd LIKE '%' || i_fms_ledger_cd || '%')
  AND (i_found IS NULL OR i_found= t1.found )
  AND (t1.status = 'A')
ORDER BY t1.fee_detail_id

RETURN 
       o_fms_ledger_cd,
       o_fee_detail_id,
       o_fee_detail_nm_en,
       o_found,
       o_fms_cd,
       o_mft_total,
       total WITH RESUME;
END FOREACH;

END PROCEDURE;
GO

CREATE  OR REPLACE PROCEDURE sp_getmttlisting(
    i_page INT, i_size INT,
    i_ss_cd_list NVARCHAR(255), -- Assuming a comma-separated list of ss_cd values
    i_orn_no NVARCHAR(20),
    i_orn_dt_fr DATETIME YEAR TO SECOND, i_orn_dt_to DATETIME YEAR TO SECOND,
    i_total_amt DECIMAL(16,2), i_order_status NVARCHAR(10),
    i_rcpt_no NVARCHAR(20), i_rcpt_dt_fr DATETIME YEAR TO SECOND, i_rcpt_dt_to DATETIME YEAR TO SECOND
)
RETURNING NVARCHAR(10) AS ss_cd,
          NVARCHAR(20) AS orn_no,
          DATETIME YEAR TO SECOND AS orn_dt,
          DECIMAL(16,2) AS total_amt,
          NVARCHAR(10) AS order_status,
          NVARCHAR(20) AS rcpt_no,
          DATETIME YEAR TO SECOND AS rcpt_dt,
          INT AS total,
          INT AS mtt_id;

DEFINE o_mtt_id INT;
DEFINE o_ss_cd NVARCHAR(10);
DEFINE o_orn_no NVARCHAR(20);
DEFINE o_total_amt DECIMAL(16,2);
DEFINE o_orn_dt DATETIME YEAR TO SECOND;
DEFINE o_order_status NVARCHAR(10);
DEFINE o_rcpt_no NVARCHAR(20);
DEFINE o_rcpt_dt DATETIME YEAR TO SECOND;
DEFINE calculatedpage INT;
DEFINE total INT;

IF (i_page IS NULL OR i_page < 1) OR (i_size IS NULL OR i_page < 1) THEN
    RAISE EXCEPTION -746, 0, "i_page or i_size is null or less than 1";
END IF;

LET calculatedpage = (i_page - 1) * i_size;

LET total = (
    SELECT COUNT(*)
    FROM (
        SELECT t1.ss_cd, t1.orn_no, t1.orn_dt, t1.total_amt, t1.order_status,
               t2.rcpt_no, t2.rcpt_dt
        FROM rms_mtt t1
        LEFT OUTER JOIN rms_mtt_rcpt t2 ON t1.mtt_id = t2.mtt_id
        WHERE (i_ss_cd_list IS NULL OR ',' || i_ss_cd_list || ',' LIKE '%,' || t1.ss_cd || ',%')
          AND (i_orn_no IS NULL OR t1.orn_no LIKE '%' || i_orn_no || '%')
          AND (i_orn_dt_fr IS NULL OR t1.orn_dt >= i_orn_dt_fr)
          AND (i_orn_dt_to IS NULL OR t1.orn_dt <= i_orn_dt_to)
          AND (i_total_amt IS NULL OR t1.total_amt = i_total_amt)
          AND (i_order_status IS NULL OR t1.order_status = i_order_status )
          AND (i_rcpt_no IS NULL OR t2.rcpt_no LIKE '%' || i_rcpt_no || '%')
          AND (i_rcpt_dt_fr IS NULL OR t2.rcpt_dt >= i_rcpt_dt_fr)
          AND (i_rcpt_dt_to IS NULL OR t2.rcpt_dt <= i_rcpt_dt_to)
        GROUP BY t1.ss_cd, t1.orn_no, t1.orn_dt, t1.total_amt, t1.order_status,
                 t2.rcpt_no, t2.rcpt_dt
    ) AS subquery
);

FOREACH c_getmtt FOR
    SELECT SKIP calculatedpage LIMIT i_size
           t1.ss_cd, t1.orn_no, t1.orn_dt, t1.total_amt, t1.order_status,
           t2.rcpt_no, t2.rcpt_dt, t1.mtt_id
    INTO o_ss_cd, o_orn_no, o_orn_dt, o_total_amt, o_order_status, o_rcpt_no, o_rcpt_dt, o_mtt_id
    FROM rms_mtt t1
    LEFT OUTER JOIN rms_mtt_rcpt t2 ON t1.mtt_id = t2.mtt_id
    WHERE (i_ss_cd_list IS NULL OR ',' || i_ss_cd_list || ',' LIKE '%,' || t1.ss_cd || ',%')
      AND (i_orn_no IS NULL OR t1.orn_no LIKE '%' || i_orn_no || '%')
      AND (i_orn_dt_fr IS NULL OR t1.orn_dt >= i_orn_dt_fr)
      AND (i_orn_dt_to IS NULL OR t1.orn_dt <= i_orn_dt_to)
      AND (i_total_amt IS NULL OR t1.total_amt = i_total_amt)
      AND (i_order_status IS NULL OR t1.order_status= i_order_status)
      AND (i_rcpt_no IS NULL OR t2.rcpt_no LIKE '%' || i_rcpt_no || '%')
      AND (i_rcpt_dt_fr IS NULL OR t2.rcpt_dt >= i_rcpt_dt_fr)
      AND (i_rcpt_dt_to IS NULL OR t2.rcpt_dt <= i_rcpt_dt_to)
    GROUP BY t1.ss_cd, t1.orn_no, t1.orn_dt, t1.total_amt, t1.order_status,
             t2.rcpt_no, t2.rcpt_dt, t1.mtt_id
RETURN o_ss_cd, o_orn_no, o_orn_dt, o_total_amt, o_order_status, o_rcpt_no, o_rcpt_dt, total, o_mtt_id WITH RESUME;
END FOREACH;

END PROCEDURE;

create procedure sp_getPGReconDetail(i_task_id nvarchar(25))
returning 
		integer as pg_txn_settlement_no,
		decimal(16,2) as pg_total_txn_settlement_amt,
        integer as pg_txn_adj_no,
		decimal(16,2) as pg_total_txn_adj_amt,
        decimal(16,2) as pg_total_txn_other,
        integer as pg_found_no,
		decimal(16,2) as pg_found_total,
        integer as pg_not_found_no,
		decimal(16,2) as pg_not_found_total,
        integer as pg_sam_no,
		decimal(16,2) as pg_sam_total,
        integer as pg_snm_no,
		decimal(16,2) as pg_snm_total,
        integer as pg_txf_no,
		decimal(16,2) as pg_txf_total,
        integer as pg_matched_no,
		decimal(16,2) as pg_matched_total,
        integer as rms_txn_no,
        integer as rms_paid_no,
		decimal(16,2) as rms_paid_total,
        integer as rms_failed_no,
		decimal(16,2) as rms_failed_total,
        integer as rms_rcpt_no,
		decimal(16,2) as rms_rcpt_total,
        integer as rms_sam_no,
		decimal(16,2) as rms_sam_total,
        integer as rms_snm_no,
		decimal(16,2) as rms_snm_total,
        integer as rms_txf_no,
		decimal(16,2) as rms_txf_total,
        integer as rms_cip_no,
		decimal(16,2) as rms_cip_total,
        integer as rms_ncp_no,
		decimal(16,2) as rms_ncp_total,
        integer as rms_nfp_no,
		decimal(16,2) as rms_nfp_total,
        nvarchar(25) as task_id,
		nvarchar(100) as recon_status,
        nvarchar(100) as dt_statement,
		nvarchar(100) as dt_settlement_char,
        nvarchar(255) as remarks,
        nvarchar(100) as task_status;

    -- payment gateway settlement summary
    define pg_txn_settlement_no integer;
    define pg_total_txn_settlement_amt decimal(16,2);
    define pg_txn_adj_no integer;
    define pg_total_txn_adj_amt decimal(16,2);
    define pg_total_txn_other decimal(16,2);

    -- discrepancy check  based on payment gateway settlement
    define pg_matched_no integer;
    define pg_matched_total decimal(16,2);
    define pg_found_no integer;
    define pg_found_total decimal(16,2);
    define pg_not_found_no integer;
    define pg_not_found_total decimal(16,2);
    define pg_sam_no integer;
    define pg_sam_total decimal(16,2);
    define pg_snm_no integer;
    define pg_snm_total decimal(16,2);
    define pg_txf_no integer;
    define pg_txf_total decimal(16,2);

    -- rms source system summary
    define rms_txn_no integer;
    define rms_paid_no integer;
    define rms_paid_total decimal(16,2);
    define rms_failed_no integer;
    define rms_failed_total decimal(16,2);

    -- discrepancy check based on rms source system
    define rms_rcpt_no integer;
    define rms_rcpt_total decimal(16,2);
    define rms_sam_no integer;
    define rms_sam_total decimal(16,2);
    define rms_snm_no integer;
    define rms_snm_total decimal(16,2);
    define rms_txf_no integer;
    define rms_txf_total decimal(16,2);
    define rms_cip_no integer;
    define rms_cip_total decimal(16,2);
    define rms_ncp_no integer;
    define rms_ncp_total decimal(16,2);
    define rms_nfp_no integer;
    define rms_nfp_total decimal(16,2);

    -- reconciliation task
    define task_id nvarchar(25);
    define recon_status nvarchar(100);
    define o_dt_statement nvarchar(100);
    define dt_settlement_char nvarchar(100);
    define o_dt_settlement date;
    define o_remarks nvarchar(255);
    define task_status nvarchar(100);

    -- get dt_settlement
    let o_dt_settlement =(select dt_settlement
                            from rms_rc_pg
                            where task_no = i_task_id);

    -- payment gateway settlement summary
    SELECT 
        SUM(CASE WHEN txn_cd = 'Settlement' THEN 1 ELSE 0 END) AS settlement_count,
        SUM(CASE WHEN txn_cd = 'Settlement' THEN txn_amt ELSE 0 END) AS settlement_sum,
        SUM(CASE WHEN txn_cd = 'Txn Adjustment' THEN 1 ELSE 0 END) AS adjustment_count,
        SUM(CASE WHEN txn_cd = 'Txn Adjustment' THEN txn_amt ELSE 0 END) AS adjustment_sum,
        SUM(CASE WHEN txn_cd NOT IN ('Settlement', 'Txn Adjustment') THEN txn_amt ELSE 0 END) AS other_sum
    into pg_txn_settlement_no,pg_total_txn_settlement_amt,pg_txn_adj_no,pg_total_txn_adj_amt,pg_total_txn_other
    FROM 
        rms_rc_pgtxn a
    left join rms_rc_pg b on a.rc_pg_id=b.rc_pg_id
    WHERE 
        b.task_no = i_task_id;

    -- discrepancy check  based on payment gateway settlement
    SELECT 
        SUM(CASE WHEN found_in_rms = 1 and txn_cd=='Settlement' THEN 1 ELSE 0 END) AS found_in_rms_1_count,
        SUM(CASE WHEN found_in_rms = 1 and txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS found_in_rms_1_sum,
        SUM(CASE WHEN found_in_rms = 0 and txn_cd=='Settlement' THEN 1 ELSE 0 END) AS found_in_rms_0_count,
        SUM(CASE WHEN found_in_rms = 0 and txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS found_in_rms_0_sum,
        SUM(CASE WHEN sub_criteria = 'SAM' and txn_cd=='Settlement' THEN 1 ELSE 0 END) AS sam_count,
        SUM(CASE WHEN sub_criteria = 'SAM' and txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS sam_sum,
        SUM(CASE WHEN sub_criteria = 'SNM' and txn_cd=='Settlement' THEN 1 ELSE 0 END) AS snm_count,
        SUM(CASE WHEN sub_criteria = 'SNM' and txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS snm_sum,
        SUM(CASE WHEN sub_criteria = 'TXF' and txn_cd=='Settlement' THEN 1 ELSE 0 END) AS txf_count,
        SUM(CASE WHEN sub_criteria = 'TXF' and txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS txf_sum,
        COUNT(*) AS total_count,
        SUM(CASE WHEN txn_cd=='Settlement' THEN txn_amt ELSE 0 END) AS total_sum
    into pg_found_no,pg_found_total,pg_not_found_no,pg_not_found_total,pg_sam_no,pg_sam_total,pg_snm_no,pg_snm_total,pg_txf_no,pg_txf_total,pg_matched_no,pg_matched_total
    FROM 
        rms_rc_pg
    INNER JOIN 
        rms_rc_pgtxn ON rms_rc_pg.rc_pg_id = rms_rc_pgtxn.rc_pg_id
    WHERE 
        rms_rc_pg.task_no = i_task_id
        and rms_rc_pgtxn.txn_cd=='Settlement';
    
    -- rms source system summary











    select COUNT(*) as total_count,
    SUM(CASE WHEN rms_mtt.order_status = 'P' THEN 1 ELSE 0 END) as paid_count,
    SUM(CASE WHEN rms_mtt.order_status = 'P' THEN pg_pymt_amt ELSE 0 END) as paid_total,
    SUM(CASE WHEN rms_mtt.order_status = 'F' THEN 1 ELSE 0 END) as fail_count,
    SUM(CASE WHEN rms_mtt.order_status = 'F' THEN pg_pymt_amt ELSE 0 END) as fail_total
    into rms_txn_no,rms_paid_no,rms_paid_total,rms_failed_no,rms_failed_total
     FROM 
            rms_mtt
        INNER JOIN 
            rms_mtt_pg ON rms_mtt.mtt_id = rms_mtt_pg.mtt_id and DATE(pymt_submit_dt)=o_dt_settlement
        INNER JOIN 
            rms_mtt_rcpt ON rms_mtt_rcpt.mtt_pg_id = rms_mtt_pg.mtt_pg_id and date(rcpt_dt)=o_dt_settlement
        INNER JOIN
            rms_rc_pgmtt ON rms_mtt_pg.mtt_pg_id = rms_rc_pgmtt.mtt_pg_id;


    -- discrepancy check based on rms source system
    SELECT 
        COUNT(*) AS total_count,
        SUM(pg_pymt_amt) AS total_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'SAM' THEN 1 ELSE 0 END) AS sam_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'SAM' THEN pg_pymt_amt ELSE 0 END) AS sam_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'SNM' THEN 1 ELSE 0 END) AS snm_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'SNM' THEN pg_pymt_amt ELSE 0 END) AS snm_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'TXF' THEN 1 ELSE 0 END) AS txf_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'TXF' THEN pg_pymt_amt ELSE 0 END) AS txf_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'CIP' THEN 1 ELSE 0 END) AS cip_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'CIP' THEN pg_pymt_amt ELSE 0 END) AS cip_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'NCP' THEN 1 ELSE 0 END) AS ncp_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'NCP' THEN pg_pymt_amt ELSE 0 END) AS ncp_sum,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'NFP' THEN 1 ELSE 0 END) AS nfp_count,
        SUM(CASE WHEN rms_rc_pgmtt.sub_criteria = 'NFP' THEN pg_pymt_amt ELSE 0 END) AS nfp_sum
    INTO rms_rcpt_no,rms_rcpt_total,rms_sam_no,rms_sam_total,rms_snm_no,rms_snm_total,rms_txf_no,rms_txf_total,rms_cip_no,rms_cip_total,rms_ncp_no,rms_ncp_total,rms_nfp_no,rms_nfp_total
    FROM 
        rms_mtt_pg
    INNER JOIN 
        rms_rc_pgmtt ON rms_rc_pgmtt.mtt_pg_id = rms_mtt_pg.mtt_pg_id
    INNER JOIN 
        rms_mtt_rcpt ON rms_mtt_rcpt.mtt_pg_id = rms_mtt_pg.mtt_pg_id
    WHERE 
        date(rms_mtt_rcpt.rcpt_dt) = o_dt_settlement;

    -- reconciliation task
    select task_no,b.nm_en,TO_CHAR(dt_statement, '%d %b %Y'),TO_CHAR(dt_settlement, '%d %b %Y'),remarks,c.nm_en
    into task_id,recon_status,o_dt_statement,dt_settlement_char,o_remarks,task_status
    from rms_rc_pg a
    left join rms_param b on a.recon_status=b.param_cd and b.param_grp_nm = 'Status-PGRC'
    left join rms_param c on a.task_status=c.param_cd and c.param_grp_nm = 'Status-Task'
    where task_no = i_task_id;

	return pg_txn_settlement_no,pg_total_txn_settlement_amt,pg_txn_adj_no,pg_total_txn_adj_amt,pg_total_txn_other,
    pg_found_no,pg_found_total,pg_not_found_no,pg_not_found_total,pg_sam_no,pg_sam_total,pg_snm_no,pg_snm_total,pg_txf_no,pg_txf_total,pg_matched_no,pg_matched_total,
    rms_txn_no,rms_paid_no,rms_paid_total,rms_failed_no,rms_failed_total,
    rms_rcpt_no,rms_rcpt_total,rms_sam_no,rms_sam_total,rms_snm_no,rms_snm_total,rms_txf_no,rms_txf_total,rms_cip_no,rms_cip_total,rms_ncp_no,rms_ncp_total,rms_nfp_no,rms_nfp_total,
    task_id,recon_status,o_dt_statement,dt_settlement_char,o_remarks,task_status;

end procedure

create procedure sp_getPGReconList(i_page integer, i_size integer, i_task_id nvarchar(25), i_file_nm nvarchar(255), i_dt_settlement_fr nvarchar(100),i_dt_settlement_to nvarchar(100), 
                            i_dt_uploaded_fr nvarchar(100),i_dt_uploaded_to nvarchar(100),
                            i_merchant_id nvarchar(25), i_task_status nvarchar(25), i_recon_status nvarchar(25))
returning 
		nvarchar(25) as i_task_id,
		nvarchar(255) as i_file_nm,
        nvarchar(100) as i_dt_settlement,
        nvarchar(100) as i_dt_uploaded,
		nvarchar(25) as i_merchant_id, 
		nvarchar(50) as i_task_status,
		nvarchar(50) as i_recon_status,
		integer as total;

    define total integer;
	define calculatedpage integer;
	define o_task_id nvarchar(25);
	define o_file_nm nvarchar(255);
	define o_dt_settlement nvarchar(100);
	define o_dt_uploaded nvarchar(100);
	define o_merchant_id nvarchar(25);
	define o_task_status nvarchar(50);
	define o_recon_status nvarchar(50);

    if (i_page is null or i_page < 1) or (i_size is null or i_page < 1) then 
        raise exception -746, 0, "i_page or i_size is null integer";
    end if

	let calculatedpage = (i_page-1)*i_size;
	let total = (select count(1)
                from rms_rc_pg a
                left join rms_rc_pgdoc b on a.rc_pg_id=b.rc_pg_id
                where (i_task_id is null or a.task_no like '%' || i_task_id || '%')
                and (i_dt_settlement_fr is null or a.dt_settlement >= to_date(i_dt_settlement_fr,'%Y-%m-%d'))
                and (i_dt_settlement_to is null or a.dt_settlement <= to_date(i_dt_settlement_to,'%Y-%m-%d'))
                and (i_dt_uploaded_fr is null or a.dt_upload >= to_date(i_dt_uploaded_fr,'%Y-%m-%d'))
                and (i_dt_uploaded_to is null or a.dt_upload <= to_date(i_dt_uploaded_to,'%Y-%m-%d'))
                and (i_merchant_id is null or a.merchant_id like '%' || i_merchant_id || '%')
                and (i_task_status is null or a.task_status=i_task_status)
                and (i_recon_status is null or a.recon_status=i_recon_status)
                and (i_file_nm is null or b.file_nm like '%' || i_file_nm || '%')
        			);

	foreach c_getripl for
        
        select skip calculatedpage limit i_size a.task_no, b.file_nm,TO_CHAR(a.dt_settlement, '%d %b %Y'),TO_CHAR(a.dt_upload, '%d %b %Y'),a.merchant_id,c.nm_en,d.nm_en
        into o_task_id, o_file_nm,o_dt_settlement,o_dt_uploaded,o_merchant_id,o_task_status,o_recon_status
        from rms_rc_pg a
        left join rms_rc_pgdoc b on a.rc_pg_id=b.rc_pg_id
        left join rms_param c on a.task_status = c.param_cd and c.param_grp_nm='Status-Task'
        left join rms_param d on a.recon_status = d.param_cd and d.param_grp_nm='Status-PGRC'
        where (i_task_id is null or a.task_no like '%' || i_task_id || '%')
        and (i_dt_settlement_fr is null or a.dt_settlement >= to_date(i_dt_settlement_fr,'%Y-%m-%d'))
        and (i_dt_settlement_to is null or a.dt_settlement <= to_date(i_dt_settlement_to,'%Y-%m-%d'))
        and (i_dt_uploaded_fr is null or a.dt_upload >= to_date(i_dt_uploaded_fr,'%Y-%m-%d'))
        and (i_dt_uploaded_to is null or a.dt_upload <= to_date(i_dt_uploaded_to,'%Y-%m-%d'))
        and (i_merchant_id is null or a.merchant_id like '%' || i_merchant_id || '%')
        and (i_task_status is null or a.task_status=i_task_status)
        and (i_recon_status is null or a.recon_status=i_recon_status)
        and (i_file_nm is null or b.file_nm like '%' || i_file_nm || '%')
        order by a.dt_upload desc
	
		return o_task_id, o_file_nm,o_dt_settlement,o_dt_uploaded,o_merchant_id,o_task_status,o_recon_status,total with resume;
	end foreach

end procedure

create procedure sp_getPGSettlementDate()
returning nvarchar (100) as o_dt_settlement;

	define o_dt_settlement nvarchar(100);
 
	foreach c_getPGSettlementDate for	
		select distinct TO_CHAR(t1.dt_settlement, '%d/%m/%Y') as formatted_date
        into o_dt_settlement
		from rms_rc_pg t1
		where t1.task_status = 'CS'
		  and t1.dt_settlement not in (
            select t2.dt_settlement
            from rms_rc_bank t1
            inner join rms_rc_pg t2 on t1.rc_pg_id = t2.rc_pg_id
            where t1.task_status <> 'CL'
          )

		return o_dt_settlement with resume;
	end foreach;

select t2.file_nm
into o_file_nm
from rms_rc_bank t1
inner join rms_rc_pgdoc t2 on t1.rc_pg_id = t2.rc_pg_id
where t1.task_no = i_task_no;

-- Total number of txn, Total Gross Amount, Total MDR, Total Net Amount
select count(t2.txn_id), sum(t2.txn_amt), sum(t2.mdr_amt), sum(t2.txn_amt) - ABS(sum(t2.mdr_amt))
into o_total_no_pg_txn, o_total_gross_amt, o_total_mdr,  o_total_net_amt
from rms_rc_bank t1
inner join rms_rc_pgtxn t2 on t1.rc_pg_id = t2.rc_pg_id
where t1.task_no = i_task_no
and t2.txn_cd = 'Settlement';

-- Bank Statement Summary
-- No of Bank stmt, All Bank Txn
select count(t2.rc_bank_id)
into o_total_no_bk_txn
from rms_rc_bankdoc t1
inner join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
where t2.task_no = i_task_no;

-- No of Bank stmt, All Bank Txn
select count(t3.rc_banktxn_id)
into o_total_bank_txn
from rms_rc_bankdoc t1
inner join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
inner join rms_rc_banktxn t3 on t1.rc_bank_id = t3.rc_bank_id
where t2.task_no = i_task_no;

-- Select PG total credit in related
select count(t1.rc_banktxn_id),sum(t1.credit) - ABS(o_total_mdr)
into o_total_pg_file_txn, o_total_pg_disbursed_amt
from rms_rc_banktxn t1
inner join rms_rc_bank t2 on t1.rc_bank_id = t2.rc_bank_id
where t2.task_no = i_task_no
and substring(t1.txn_ref from 0 for 14) in (select stmt_no 
                                            from rms_rc_pg t1
                                            inner join rms_rc_bank t2 on t1.rc_pg_id = t2.rc_pg_id);
--and t1.status = 'SBSR';

-- Recon Bank
-- Select rc_bank
select t1.task_no, t4.nm_en, t3.nm_en, t1.remarks, t2.stmt_no, t2.dt_settlement
into o_task_no, o_recon_status, o_task_status, o_remarks, o_stmt_no, o_dt_settlement
from rms_rc_bank t1
inner join rms_rc_pg t2 on t1.rc_pg_id = t2.rc_pg_id
inner join rms_param t3 on t1.task_status = t3.param_cd and t3.param_grp_nm = 'Status-Task'
inner join rms_param t4 on t1.recon_status = t4.param_cd and t4.param_grp_nm = 'Status-BRC'
where t1.task_no = i_task_no;

commit work;

return o_file_nm, o_total_no_pg_txn, o_total_gross_amt, o_total_mdr, o_total_net_amt,
        o_total_no_bk_txn, o_total_bank_txn, o_total_pg_file_txn, o_total_pg_disbursed_amt, 
         o_task_no, o_recon_status, o_task_status, o_remarks, o_stmt_no, o_dt_settlement;

end procedure;

CREATE PROCEDURE sp_getrunno(i_runno_cd NVARCHAR(50))
RETURNING NVARCHAR(20) AS running_no;

DEFINE o_run_no NVARCHAR(20);
DEFINE o_dt_last_runno DATE;
DEFINE o_day_chk INT;

BEGIN WORK;

/* Reset running no */
SELECT FIRST 1 dt_last_runno
INTO o_dt_last_runno
FROM rms_runno
WHERE runno_cd = i_runno_cd;

IF DAY(TODAY) <> DAY(o_dt_last_runno)
	THEN 	UPDATE 	rms_runno
			SET		last_runno = 0,
					dt_last_runno = current
			WHERE 	runno_cd = i_runno_cd;
END IF;

/* UPDATE last running no */
UPDATE rms_runno
SET last_runno = last_runno +1
WHERE runno_cd = i_runno_cd;

/* Select running no */
SELECT FIRST 1 (i_runno_cd || TO_CHAR(TODAY, '%Y%m%d') || LPAD(last_runno, 6,0)) as run_no
INTO o_run_no
FROM rms_runno
WHERE runno_cd = i_runno_cd;

COMMIT WORK;

RETURN o_run_no ;

END PROCEDURE

CREATE PROCEDURE sp_getuseranduserroles(
    i_page INTEGER,
    i_size INTEGER,
    i_user NVARCHAR(50),
    i_user_role NVARCHAR(50))
RETURNING 
    NVARCHAR(50) AS ssm4uuserrefno,
    NVARCHAR(50) AS name,
    NVARCHAR(50) AS email,
    NVARCHAR(25) AS created_by,
    NVARCHAR(25) AS modified_by,
    NVARCHAR(10) AS status,
    LVARCHAR(MAX) AS roles,
    INTEGER AS total_roles;

DEFINE o_user NVARCHAR(50);
DEFINE o_name NVARCHAR(50);
DEFINE o_email NVARCHAR(50);
DEFINE o_created_by NVARCHAR(25);
DEFINE o_modified_by NVARCHAR(25);
DEFINE o_status NVARCHAR(10);
DEFINE o_roles LVARCHAR(MAX);
DEFINE total_roles INTEGER;
DEFINE calculatedpage INTEGER;

IF (i_page IS NULL OR i_page < 1) THEN 
    RAISE EXCEPTION -746, 0, "i_page is null or invalid";
END IF;

LET calculatedpage = (i_page - 1) * i_size;

LET total_roles = (SELECT COUNT(DISTINCT u.ssm4uuserrefno)
                   FROM rms_user u
                   LEFT OUTER JOIN rms_user_role ur ON u.ssm4uuserrefno = ur.ssm4uuserrefno
                   LEFT OUTER JOIN rms_role r ON ur.role_id = r.role_id
                   WHERE (i_user IS NULL OR u.nm LIKE '%' || i_user || '%')
                     AND (i_user_role IS NULL OR r.role_nm_en = i_user_role));

FOREACH c_getuserroles FOR
    SELECT SKIP calculatedpage LIMIT i_size
        u.ssm4uuserrefno,
        u.nm,
        u.email,
        u.created_by,
        u.modified_by,
        u.status,
        sp_GROUP_CONCAT(r.role_nm_en) AS roles
    INTO 
        o_user,
        o_name,
        o_email,
        o_created_by,
        o_modified_by,
        o_status,
        o_roles
    FROM rms_user u
    LEFT OUTER JOIN rms_user_role ur ON u.ssm4uuserrefno = ur.ssm4uuserrefno
    LEFT OUTER JOIN rms_role r ON ur.role_id = r.role_id
    WHERE (i_user IS NULL OR u.nm LIKE '%' || i_user || '%')
    AND (i_user_role IS NULL OR r.role_nm_en = i_user_role OR u.ssm4uuserrefno IN (
        SELECT ur_inner.ssm4uuserrefno
        FROM rms_user_role ur_inner
        JOIN rms_role r_inner ON ur_inner.role_id = r_inner.role_id
        WHERE r_inner.role_nm_en = i_user_role
    ))
    GROUP BY u.ssm4uuserrefno, u.nm, u.email, u.created_by, u.modified_by, u.status
    order by u.nm
    RETURN o_user, o_name, o_email, o_created_by, o_modified_by, o_status, o_roles, total_roles WITH RESUME;
END FOREACH;

END PROCEDURE;

CREATE PROCEDURE sp_getwfh_ast(i_task_id NVARCHAR(50), i_status NVARCHAR(20))
RETURNING NVARCHAR(20) AS assign_to,
          NVARCHAR(50) AS assign_to_nm;

DEFINE o_assign_to 		NVARCHAR(20);
DEFINE o_assign_to_nm 	NVARCHAR(50);
DEFINE o_wf_id 			INT8;
define pairStatus       NVARCHAR(20);

LET o_wf_id = TO_NUMBER(SUBSTR(i_task_id, 3));

IF i_status = 'P-RHOD' THEN
    LET pairStatus = 'Q-RHOD';
elif i_status = 'P-FA' THEN
    LET pairStatus = 'Q-FA';
elif i_status = 'Q-FA' THEN
    LET pairStatus = 'P-FA';
elif i_status = 'Q-RHOD' THEN
    LET pairStatus = 'P-RHOD';
ELSE
    LET pairStatus = '';
END IF;

BEGIN WORK;

    --SELECT DISTINCT t1.assign_to, t2.nm
    --INTO o_assign_to, o_assign_to_nm
    --FROM rms_mft_wfhist t1
    --LEFT OUTER JOIN rms_user t2 ON t1.assign_to = t2.ssm4uuserrefno
    --WHERE t1.wf_id = o_wf_id
    --AND t1.status = i_status;
IF i_status IN ('Q-R','P-FHOD') THEN
 	  SELECT assign_to, nm 
 	  INTO o_assign_to, o_assign_to_nm
 	  FROM (
      SELECT ROW_NUMBER () OVER (PARTITION BY t1.wf_id ORDER BY t1.wf_id, t1.dt_activity DESC) AS rn, t1.assign_to, t2.nm FROM rms_mft_wfhist t1
	  LEFT OUTER JOIN rms_user t2 ON t1.assign_to = t2.ssm4uuserrefno
      WHERE t1.wf_id = o_wf_id
      AND t1.status = i_status 
      ) T1 where rn = 1;
ELSE
	  SELECT assign_to, nm 
 	  INTO o_assign_to, o_assign_to_nm
 	  FROM (
      SELECT ROW_NUMBER () OVER (PARTITION BY t1.wf_id ORDER BY t1.wf_id, t1.dt_activity DESC) AS rn, t1.assign_to, t2.nm FROM rms_mft_wfhist t1
	  LEFT OUTER JOIN rms_user t2 ON t1.assign_to = t2.ssm4uuserrefno
      WHERE t1.wf_id = o_wf_id
      AND (t1.status = i_status or t1.status = pairStatus) 
      ) T1 where rn = 1;
END IF;

COMMIT WORK;

RETURN o_assign_to, o_assign_to_nm;
                        
END PROCEDURE;

CREATE PROCEDURE sp_getwfh_status(i_task_id NVARCHAR(50), i_status NVARCHAR(20))
RETURNING NVARCHAR(20) AS assign_to,
          NVARCHAR(50) AS assign_to_nm,
          NVARCHAR(10) AS status;

DEFINE o_assign_to 		NVARCHAR(20);
DEFINE o_assign_to_nm 	NVARCHAR(50);
DEFINE o_status 		NVARCHAR(10);
DEFINE o_wf_id 			INT8;

LET o_wf_id = TO_NUMBER(SUBSTR(i_task_id, 3));

BEGIN WORK;


 	  SELECT assign_to, nm, status
 	  INTO o_assign_to, o_assign_to_nm, o_status 
 	  FROM (
      SELECT ROW_NUMBER () OVER (PARTITION BY t1.wf_id ORDER BY t1.wf_id, t1.dt_activity DESC) AS rn, t1.assign_to, t2.nm, t1.status
      FROM rms_mft_wfhist t1
	  LEFT OUTER JOIN rms_user t2 ON t1.assign_to = t2.ssm4uuserrefno
      WHERE t1.wf_id = o_wf_id
      AND t1.status != i_status
      ) T1 where rn = 1;

COMMIT WORK;

RETURN o_assign_to, o_assign_to_nm, o_status;
                        
END PROCEDURE;

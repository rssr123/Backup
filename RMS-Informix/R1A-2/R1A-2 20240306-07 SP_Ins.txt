CREATE or REPLACE PROCEDURE sp_insbank(i_file_nm nvarchar(255), i_file_content BLOB, i_file_type nvarchar(5), i_file_size_kb INT,
							i_dt_settlement nvarchar(100), i_created_by nvarchar(25))
                            
returning int as result;											

define o_result int;
define o_exist int;
define o_task_no nvarchar(25);
define o_rc_bank_id nvarchar(20);
define o_rc_pg_id nvarchar(20);
define o_exist2 int;

on exception
	rollback work;
end exception;

-- Set a default value for o_result
let o_result = 0;

-- check exist settlement's status
select count(1), rc_pg_id
into o_exist, o_rc_pg_id
from rms_rc_pg
where dt_settlement = to_date(i_dt_settlement,'%d/%m/%Y')
and task_status = 'CS'
group by rc_pg_id;

if o_exist > 0 then

select count(1)
into o_exist2
from rms_rc_bank
where rc_pg_id = o_rc_pg_id
and task_status = 'O';

    if o_exist2 = 0 then

        -- get latest running no
        EXECUTE PROCEDURE sp_getrunno('RC') INTO o_task_no;

        BEGIN WORK;
        set lock mode to wait 30;

        -- insert rms_rc_bank
        INSERT INTO rms_rc_bank(task_no, rc_pg_id, task_status, recon_status, dt_created, dt_modified, created_by, modified_by, dt_upload) 
                         VALUES(o_task_no, o_rc_pg_id, "O", "PE", current, current, i_created_by, i_created_by, current);

        -- get inserted id
        let o_rc_bank_id = DBINFO('Serial8');

        -- insert rms_rc_pgdoc
        INSERT INTO rms_rc_bankdoc( rc_bank_id, file_nm, file_content, file_type, file_size_kb, dt_created, dt_modified, created_by, modified_by, status) 
                            VALUES( o_rc_bank_id, i_file_nm, i_file_content, i_file_type, i_file_size_kb, current, current, i_created_by, i_created_by, 'A');

        -- get inserted id
        let o_result = DBINFO('Serial8');

        COMMIT WORK;    

    end if;
end if;

if o_result > 0 then
    return o_result;
else
    return -1;
end if;

end procedure

create procedure sp_insdi(i_fee_detail_id nvarchar(10), i_txn_type nvarchar(10), i_entity_no nvarchar(20), i_entity_type nvarchar(1), 
                            i_dt_effective date, i_dt_expiry date, i_item_ref_no nvarchar(30),
                            i_approval_status nvarchar(10), i_dt_approval date)
returning int8 as result;

define o_exist int;
define o_exist_a int;
define o_fee_detail_pk int;
define o_qty int;
define o_unit_fee decimal(16,2);
define o_dt_effective date;
define o_result int;
define o_total_fee decimal(16,2);
define o_bal_amt decimal(16,2);

on exception
	rollback work;
end exception;

let o_result = 0;

-- Record exists checking
select count(1)
into o_exist
from rms_mtt_item
where fee_detail_id = i_fee_detail_id
and entity_type = i_entity_type
and entity_no = i_entity_no
and item_ref_no = i_item_ref_no;

if o_exist > 0 then

    begin work;
    set lock mode to wait 30;

    -- Select from rms_mtt_item
    select first 1 fee_detail_pk, qty, unit_fee
    into o_fee_detail_pk, o_qty, o_unit_fee
    from rms_mtt_item
    where fee_detail_id = i_fee_detail_id
    and entity_type = i_entity_type
    and entity_no = i_entity_no
    and item_ref_no = i_item_ref_no;

    select count(1)
    into o_exist_a
    from rms_di
    where fee_detail_id = i_fee_detail_id
    and entity_type = i_entity_type
    and entity_no = i_entity_no
    and item_ref_no = i_item_ref_no
    and status not in ('C','T');
    
    let o_total_fee = o_qty * o_unit_fee;

    if o_exist_a > 0 then

        -- Insert into rms_di
        insert into rms_di(fee_detail_pk, fee_detail_id, entity_type, entity_no, dt_effective, dt_expiry,
                            dt_termination, item_ref_no, approval_status, dt_approval, no_of_yr, unit_fee, total_fee,
                                bal_di_amt, next_calc_dt, dt_created, dt_modified, created_by, modified_by, status, txn_type) 
        values(o_fee_detail_pk, i_fee_detail_id, i_entity_type, i_entity_no, i_dt_effective, i_dt_expiry,
             '', i_item_ref_no, i_approval_status, i_dt_approval, o_qty, o_unit_fee, o_total_fee,
                o_total_fee, i_dt_effective, current, current, 'system', 'system', 'R', i_txn_type);
    else
        -- Amount Calculation
        let o_bal_amt = o_total_fee - o_unit_fee;
        let o_dt_effective = EXTEND(EXTEND(i_dt_effective, YEAR TO DAY) + 1 UNITS YEAR, YEAR TO DAY);

        -- Insert into rms_di
        insert into rms_di(fee_detail_pk, fee_detail_id, entity_type, entity_no, dt_effective, dt_expiry,
                            dt_termination, item_ref_no, approval_status, dt_approval, no_of_yr, unit_fee, total_fee,
                                bal_di_amt, next_calc_dt, dt_created, dt_modified, created_by, modified_by, status, txn_type) 
        values(o_fee_detail_pk, i_fee_detail_id, i_entity_type, i_entity_no, i_dt_effective, i_dt_expiry,
             '', i_item_ref_no, i_approval_status, i_dt_approval, o_qty, o_unit_fee, o_total_fee,
                o_bal_amt, o_dt_effective, current, current, 'system', 'system', 'R', i_txn_type);
    end if

    -- Set rows effected in variable
    let o_result = dbinfo('serial8');

    -- Insert into rms_di audit
    insert into rms_di_a(di_id, fee_detail_pk, fee_detail_id, dt_txn, action_type, 
                            entity_type, entity_no, bal_di_amt_b4, bal_di_amt_af, status_b4, status_af, 
                                dt_created, dt_modified, created_by, modified_by, status) 
    values(o_result, o_fee_detail_pk, i_fee_detail_id, current, 'PCR',
                i_entity_type, i_entity_no, 0.00, o_total_fee, '', 'R',
                    current, current, 'system', 'system', 'R');

    -- Insert into rms_di journal debit

    -- Insert into rms_di journal credit

    commit work;

end if;


if o_result > 0 then
    return o_result;
else
    let o_result = 0;
    return o_result;
end if;

end procedure

create procedure sp_insdi_tmn_log(i_txn_type nvarchar(10), i_entity_no nvarchar(20), i_entity_type nvarchar(1),
                                    i_dt_termination date, i_dt_approval date)
returning int8 as result;

define o_exists int;
define o_di_id int8;
define o_fee_detail_pk int;
define o_fee_detail_id nvarchar(10);
define o_entity_type nvarchar(1);
define o_entity_no nvarchar(20);
define o_total_fee decimal(16,2);
define o_bal_di_amt decimal(16,2);
define o_status nvarchar(10);
define o_modified_by nvarchar(25);

on exception
	rollback work;
end exception;

begin work;
set lock mode to wait 10;

select count(1)
into o_exists
from rms_di
where entity_type = i_entity_type
and entity_no = i_entity_no
and i_dt_termination between dt_effective and dt_expiry
and status not in ('C','T');

if o_exists > 0 then

    -- Select di_id from rms_di
    select first 1 di_id, fee_detail_pk, fee_detail_id, entity_type, entity_no, total_fee, bal_di_amt, status, modified_by
    into o_di_id, o_fee_detail_pk, o_fee_detail_id, o_entity_type, o_entity_no, o_total_fee, o_bal_di_amt, o_status, o_modified_by
    from rms_di
    where entity_type = i_entity_type
    and entity_no = i_entity_no
    and i_dt_termination between dt_effective and dt_expiry
    and status not in ('C','T');

    -- Insert statement into rms_di_tmn_log
    insert into rms_di_tmn_log(di_id, entity_type, entity_no, dt_termination, 
                                dt_created, dt_modified, created_by, modified_by, status, txn_type) 
    values(o_di_id, i_entity_type, i_entity_no, i_dt_termination,
               current, current, 'system', 'system', o_status, i_txn_type);
    
    -- Update statement in rms_di
    update rms_di
    set dt_termination = i_dt_termination,
        status = 'T',
        dt_modified = current
    where entity_type = i_entity_type
    and entity_no = i_entity_no
    and i_dt_termination between dt_effective and dt_expiry
    and status not in ('C','T');

    -- Insert statement into rms_di audit 
    insert into rms_di_a(di_id, fee_detail_pk, fee_detail_id, dt_txn, action_type, 
                            entity_type, entity_no, bal_di_amt_b4, bal_di_amt_af, status_b4, status_af, 
                                dt_created, dt_modified, created_by, modified_by, status) 
    values(o_di_id, o_fee_detail_pk, o_fee_detail_id, current, 'PCT',
                o_entity_type, o_entity_no, o_total_fee, o_bal_di_amt, o_status, 'T',
                    current, current, o_modified_by, o_modified_by, o_status);

    -- Insert statement into rms_di journal debit

    -- Insert statement into rms_di journal credit

else

    let o_di_id = -1;

end if;

commit work;

return o_di_id;

end procedure

CREATE PROCEDURE informix.sp_insfmsledgerdoc (i_fms_id INT8, i_file_nm NVARCHAR(255), i_file_content BLOB, i_file_type NVARCHAR(5), i_file_size INT,
											i_created_by NVARCHAR(25), i_modified_by NVARCHAR(25), i_status NVARCHAR(10), i_fee_detail_id NVARCHAR(10),
                                            i_fee_detail_nm_en NVARCHAR(100), i_fms_ledger_cd NVARCHAR(50))
RETURNING INT AS result,
            INT AS file_index;									

    DEFINE o_result INT;
    DEFINE temp_count INT;
    DEFINE t_file_index INT;
    DEFINE temp_count_2 INT;
    DEFINE i_file_size_kb INT;

    
    SELECT COUNT(*)
    INTO temp_count
    FROM rms_fms_doc
    WHERE fms_id = i_fms_id
    AND file_nm = i_file_nm;

    LET i_file_size_kb = i_file_size / 1024;

    -- Insert into rms_fms_doc


    IF temp_count > 0 THEN
        -- File with the same name already exists, update the existing record
        UPDATE rms_fms_doc 
        SET file_content = i_file_content,
            file_type = i_file_type,
            file_size_kb = i_file_size_kb,
            dt_modified = CURRENT,
            modified_by = i_modified_by,
            status = i_status
        WHERE fms_id = i_fms_id
          AND file_nm = i_file_nm;
    ELSE
        INSERT INTO rms_fms_doc
            (fms_id, file_nm, file_content, file_type, file_size_kb, dt_created, dt_modified, created_by, modified_by, status)
        VALUES 
            (i_fms_id, i_file_nm, i_file_content, i_file_type, i_file_size_kb, current, current, i_created_by, i_modified_by, i_status);
    END IF;

    SELECT file_index INTO t_file_index FROM rms_fms_doc WHERE file_nm=i_file_nm and file_content = i_file_content and i_fms_id = fms_id;


    -- Check if the record with i_fee_detail_id and i_fms_ledger_cd already exists
    SELECT COUNT(*) INTO o_result
    FROM rms_fms_ledger
    WHERE fee_detail_id = i_fee_detail_id
    AND fms_id = i_fms_id
    AND file_index <> t_file_index
    AND status = 'A';

    IF o_result > 0 THEN

        UPDATE rms_fms_ledger
        SET
            status = 'D'
        WHERE fee_detail_id = i_fee_detail_id
        AND fms_id = i_fms_id
        AND file_index <> t_file_index  
        AND status = 'A';
    END IF;

        -- If the record doesn't exist, insert it
        INSERT INTO rms_fms_ledger
            (fms_id, fee_detail_id, fee_detail_nm_en, fms_ledger_cd, dt_created, dt_modified, created_by, modified_by, file_index, status)
        VALUES 
            (i_fms_id, i_fee_detail_id, i_fee_detail_nm_en, i_fms_ledger_cd, current, current, i_created_by, i_modified_by, t_file_index, 'A');



        UPDATE rms_fms_ledger t1
SET found = 
    CASE 
        WHEN EXISTS (
            SELECT 1 
            FROM rms_fms_ledger t2 
            WHERE t2.fee_detail_id = t1.fee_detail_id
            AND t2.file_index = t1.file_index
            AND t2.fms_id = t1.fms_id -- Exclude the current record
            AND t2.fms_ledger_id <> t1.fms_ledger_id
            AND t2.status = 'A'
        ) THEN 'DU'  -- Duplicate
        WHEN EXISTS (
            SELECT 1 
            FROM rms_mft t3 
            WHERE t3.fee_detail_id = t1.fee_detail_id
            AND t1.status = 'A'
        ) THEN 'F'  -- Not Found
        ELSE 'NF'     -- Found
    END;
    
        -- If the record exists, update it
        

    LET o_result = DBINFO('sqlca.sqlerrd2');

    RETURN o_result, t_file_index;

END PROCEDURE;
GO

CREATE PROCEDURE sp_inspayment(
	i_mtt_id INT,

	i_pg_pymt_amt DECIMAL,

	i_username_m NVARCHAR(25)
	)
 
	RETURNING 	NVARCHAR(20) AS pymt_id,
				NVARCHAR(7) AS transaction_type,

				NVARCHAR(20) AS ord_no,
				NVARCHAR(100) AS pymt_desc,
				NVARCHAR(1) AS return_url,
				DECIMAL(16,2) AS amt,
				NVARCHAR(3) AS curr_cd,
				NVARCHAR(45) AS cust_ip,
				NVARCHAR(255) AS cust_nm,
				NVARCHAR(15) AS cust_ph,
				NVARCHAR(1) AS hash_value,
				NVARCHAR(255) AS page_timeout;
 
	DEFINE o_pymt_id NVARCHAR(20);
	DEFINE o_pg_txn_type NVARCHAR(7);
	DEFINE o_pg_pymt_method NVARCHAR(3);
	DEFINE o_pg_service_id NVARCHAR(3);
	DEFINE o_orn_no NVARCHAR(20);
	DEFINE o_pg_pymt_desc NVARCHAR(100);
	DEFINE o_pg_pymt_amt DECIMAL(16,2);
	DEFINE o_pg_curr_cd NVARCHAR(3);
	DEFINE o_cust_ip NVARCHAR(45);
	DEFINE o_cust_nm NVARCHAR(255);
	DEFINE o_cust_ph NVARCHAR(15);
	DEFINE o_page_timeout NVARCHAR(255);
	DEFINE o_return_url NVARCHAR(1);
	DEFINE o_hash_value NVARCHAR(1);
	DEFINE o_f_pymt_id NVARCHAR(20);

EXECUTE PROCEDURE sp_getrunno('PG') INTO o_f_pymt_id;

BEGIN WORK; 
 
 	LET o_return_url = '';
 	LET o_hash_value = '';
 
	INSERT INTO rms_mtt_pg (
    	mtt_id, pymt_submit_dt, pg_txn_type, pg_pymt_method, pg_service_id, pg_pymt_desc, pg_pymt_amt,
    	pg_curr_cd, pg_lang_cd, pg_txn_status,dt_created, dt_modified, created_by, modified_by,pg_pymt_id
	)
	VALUES (
	    i_mtt_id, current, 'SALE', i_pg_pymt_method, i_pg_service_id, 'Payment to Suruhanjaya Syarikat Malaysia',
	    i_pg_pymt_amt, 'MYR', i_pg_lang_cd, 2,CURRENT, CURRENT, i_username_c, i_username_m,o_f_pymt_id
	); -- default pg_txn_status = 2
 








	SELECT FIRST 1 config_value
	INTO o_page_timeout
	FROM rms_config
	WHERE config_cd = 'PIP_BlockPayOnline_In_Minute';
	
	/*Return last inserted record row data*/
	SELECT FIRST 1 t1.pg_txn_type, t1.pg_pymt_method ,t1.pg_service_id, t2.orn_no, t1.pg_pymt_desc, t1.pg_pymt_amt, t1.pg_curr_cd, t2.cust_ip, t2.cust_nm, t2.cust_phone
	INTO o_pg_txn_type, o_pg_pymt_method, o_pg_service_id, o_orn_no, o_pg_pymt_desc, o_pg_pymt_amt, o_pg_curr_cd, o_cust_ip, o_cust_nm, o_cust_ph
	FROM rms_mtt_pg t1
	LEFT OUTER JOIN rms_mtt t2 ON t1.mtt_id = t2.mtt_id
	WHERE t1.pg_pymt_id = o_f_pymt_id;

COMMIT WORK;
	
RETURN o_f_pymt_id,o_pg_txn_type, o_pg_pymt_method, o_pg_service_id, o_orn_no, o_pg_pymt_desc, o_return_url,
	o_pg_pymt_amt, o_pg_curr_cd, o_cust_ip, o_cust_nm, o_cust_ph, o_hash_value,o_page_timeout;

END PROCEDURE;

	i_cust_ip nvarchar(45),i_cust_nm nvarchar(255),i_cust_addr_1 nvarchar(155),i_cust_addr_2 nvarchar(155),i_cust_addr_3 nvarchar(155),
	i_cust_postcode nvarchar(15),i_cust_city nvarchar(50),i_cust_state  nvarchar(50),i_cust_email nvarchar(155),i_cust_phone nvarchar(15),
	i_total_amt decimal,i_ss_return_url lvarchar(2000),i_username_c nvarchar(25), i_username_m nvarchar(25))
 
	RETURNING INT AS mtt_id;
	
	DEFINE o_mtt_id INT;
    DEFINE o_orn_no_exists INT;
    DEFINE o_chckpnt INT;
    
    BEGIN WORK;
    
    SELECT COUNT(1)
    INTO o_orn_no_exists
    FROM rms_mtt
    WHERE orn_no = UPPER(i_orn_no);
    IF o_orn_no_exists = 0 THEN
        INSERT INTO rms_mtt (
            rms_type, ss_cd, orn_no, orn_dt, cust_ip, cust_nm, cust_addr_1, cust_addr_2, cust_addr_3, cust_postcode, 
            cust_city, cust_state, cust_email, cust_phone, total_amt, ss_return_url,order_status , dt_created, dt_modified, 
            created_by, modified_by
        )
        VALUES (
            i_rms_type, i_ss_cd, UPPER(i_orn_no), i_orn_dt, i_cust_ip, i_cust_nm, i_cust_addr_1, i_cust_addr_2, i_cust_addr_3, i_cust_postcode, 
            i_cust_city, i_cust_state, i_cust_email, i_cust_phone, i_total_amt, i_ss_return_url,'PIP', current, current, i_username_c, i_username_m
        );
        LET o_mtt_id = DBINFO('serial8');
        LET o_chckpnt = 1;
	    --RETURN o_mtt_id;
    ELSE
        LET o_chckpnt = 2;

        UPDATE rms_mtt
        SET order_status = 'PIP',
        dt_modified = current,
        modified_by=i_username_m
        WHERE orn_no = UPPER(i_orn_no);

        select mtt_id 
        into o_mtt_id
        from rms_mtt 
        WHERE orn_no = UPPER(i_orn_no);
        --RETURN -1;
    END IF;

COMMIT WORK;

IF o_chckpnt = 1 or o_chckpnt = 2  THEN
    RETURN o_mtt_id;
ELSE
    RETURN 0;
END IF;
    










END PROCEDURE

CREATE PROCEDURE sp_inspgmtttxn(i_rc_pg_id int8) -- to get settlement date
returning int8 as result;
 
define o_result int;
define o_exist int;
define o_compare_range int;
define o_dt_settlement date;

on exception
	rollback work;
end exception;
 
-- Set a default value
LET o_result = '0';
 
begin work;
set lock mode to wait 30;

-- create a temporary table
create temp table temppgmtt (
    rc_pg_id int8,
    mtt_pg_id int8,
    found_in_pg int,
    sub_criteria nvarchar(10)
);
 
-- check exist
select count(1)
into o_exist
from rms_rc_pg
where rc_pg_id = i_rc_pg_id;
 
if o_exist > 0 then

    -- get settlement date
    select dt_settlement
    into o_dt_settlement
    from rms_rc_pg
    where rc_pg_id = i_rc_pg_id;

    -- get compare range
    select config_value
    into o_compare_range
    from rms_config
    where config_cd='PGRecon_Period_In_Days';

    --select statement
    insert into temppgmtt (mtt_pg_id,rc_pg_id, sub_criteria,found_in_pg)
    SELECT 
        rms_mtt_pg.mtt_pg_id, 
        rms_rc_pgtxn.rc_pg_id,
        CASE 
            WHEN rms_rc_pgtxn.txn_id IS NULL THEN 'NF' 
            WHEN rms_mtt_pg.pg_txn_status = 1 THEN 'TXF' 
            WHEN rms_mtt_pg.pg_pymt_amt = rms_rc_pgtxn.txn_amt THEN 'SAM' 
            ELSE 'SNM' 
        END AS status,
        CASE 
            WHEN rms_rc_pgtxn.txn_id IS NULL THEN 0 
            ELSE 1 
        END AS found_in
    FROM 
        rms_mtt_pg 
    LEFT JOIN 
        rms_rc_pgtxn 
    ON 
        rms_mtt_pg.pg_txn_id = rms_rc_pgtxn.txn_id 
    WHERE rms_mtt_pg.pg_txn_id is not null and rms_rc_pgtxn.txn_cd='Settlement'
    and DATE(rms_mtt_pg.pymt_submit_dt) = o_dt_settlement;
    --and DATE(rms_mtt_pg.pymt_submit_dt) <= o_dt_settlement and DATE(rms_mtt_pg.pymt_submit_dt) >= (o_dt_settlement -o_compare_range UNITS DAY);

    -- insert rms_rc_pgmtt
    INSERT INTO informix.rms_rc_pgmtt( rc_pg_id, mtt_pg_id, found_in_pg, sub_criteria, 
                                        dt_created, dt_modified, created_by, modified_by, status) 
    select rc_pg_id, mtt_pg_id, found_in_pg, sub_criteria,current, current, 'system', 'system', 'A'
    from temppgmtt;
 
    -- get inserted id
    SELECT DBINFO('Serial8') 
	INTO o_result
	FROM dual;

    -- update rms_rc_pg'recon_status
    update rms_rc_pg
    set recon_status ='PSSCR'
    where rc_pg_id =i_rc_pg_id;

drop table temppgmtt;
 
end if;
 
commit work;
 
if o_result > 0 then
    return o_result;
else
    return -1;
end if;
 
end procedure

CREATE PROCEDURE sp_inspgtxn(i_rc_pg_id int8,i_dt_txn NVARCHAR(100),i_txn_id NVARCHAR(50),i_txn_type NVARCHAR(25),i_txn_cd NVARCHAR(25),
                            i_txn_amt decimal(16,2),i_mdr_amt decimal(16,2),i_sst_amt decimal(16,2),i_net_amt decimal(16,2),
                            i_created_by NVARCHAR(25),i_modified_by NVARCHAR(25))
returning int8 as result;
 
define o_result int;
define o_exist int;
define o_exist_in_rms int;
define o_mtt_pg_status nvarchar(25);
define o_mtt_pg_amt decimal(16,2);
define o_amt_match int;
define o_sub_cri NVARCHAR(10);
 
on exception
	rollback work;
end exception;
 
-- Set a default value
LET o_result = '0';
LET o_mtt_pg_status = '';
LET o_mtt_pg_amt=0;
LET o_exist_in_rms=0;
LET o_sub_cri='NFR';
 
begin work;
set lock mode to wait 30;
 
-- check exist
select count(1)
into o_exist
from rms_rc_pg
where rc_pg_id = i_rc_pg_id;
 
if o_exist > 0 then
 
    -- check the txnID exist in rms
    select count(1)
    into o_exist_in_rms
    from rms_mtt_pg
    where pg_txn_id=SUBSTR(i_txn_id, 9)
    and i_txn_cd=='Settlement';
 
    if o_exist_in_rms >0 then
 
        -- check the txnID exist in rms
        select pg_txn_status,pg_pymt_amt
        into o_mtt_pg_status,o_mtt_pg_amt
        from rms_mtt_pg
        where pg_txn_id=SUBSTR(i_txn_id, 9);
 
    end if;
 
    -- if the payment status = Failed
    if o_mtt_pg_status=='1' then
        let o_sub_cri='TXF';
    end if;
 
    -- if the payment status = Paid
    if o_mtt_pg_status=='0' then
 
        -- if the payment amount is same
        if o_mtt_pg_amt == i_txn_amt then
            let o_sub_cri='SAM';
        end if;
 
        -- if the payment amoount is not same
        if o_mtt_pg_amt <> i_txn_amt then
            let o_sub_cri='SNM';
        end if;
 
    end if;
 
    -- insert rms_rc_pg_txn
    INSERT INTO rms_rc_pgtxn(rc_pg_id, dt_txn, txn_id, txn_type, txn_cd, 
                            txn_amt, mdr_amt, sst_amt, net_amt, dt_created, 
                            dt_modified, created_by, modified_by, status, found_in_rms, 
                            sub_criteria) 
	VALUES(i_rc_pg_id, TO_DATE(i_dt_txn, '%m/%d/%Y %I:%M:%S %p'), SUBSTR(i_txn_id, 9),i_txn_type, i_txn_cd,
            i_txn_amt, i_mdr_amt, i_sst_amt, i_net_amt, current, 
            current, i_created_by,i_modified_by,'A', o_exist_in_rms,  
            o_sub_cri);
 
    -- get inserted id
    SELECT DBINFO('Serial8') 
	INTO o_result
	FROM dual;
 
end if;
 
commit work;
 
if o_result > 0 then
    return o_result;
else
    return -1;
end if;
 
end procedure
GO

CREATE PROCEDURE sp_insuserrole(userId nvarchar(255), roleId integer)
	returning int as row_count;
 
	Define row_count int;
 
	INSERT INTO rms_user_role(ssm4uuserrefno, role_id)
		VALUES
		(userId, roleId);

	SELECT COUNT(1)
	INTO row_count
	FROM rms_user_role
	WHERE ssm4uuserrefno = userId and role_id = roleId and status = 'A';

	RETURN row_count;

END PROCEDURE;

CREATE OR REPLACE procedure sp_updricp(i_entity_type nvarchar(1), i_entity_no nvarchar(20), i_cp_no nvarchar(20),
i_status nvarchar(10), i_audit_action nvarchar(10), i_old_status nvarchar(10), i_modified_by nvarchar(25))

	returning int8 as result;

	define o_exist int;
	define o_result int;
	define o_accr_amt decimal;

	on exception
		rollback work;
	end exception;
	
	begin work;
	set lock mode to wait 30;

	-- Exist checking
	select ricp_id, accr_amt
	into o_exist, o_accr_amt
	from rms_ricp
	where entity_type = i_entity_type
	and entity_no = i_entity_no
	and cp_no = i_cp_no
	and status = i_old_status;

	let o_result = 0;

	if o_exist > 0 then
	
		update rms_ricp
		set status=i_status,
		dt_modified = current,
		modified_by = i_modified_by
	    where ricp_id = o_exist;
	   
	   	insert into rms_ricp_a
		(ricp_id,dt_txn,action_type,cp_no,entity_type,entity_no,accr_amt_b4,accr_amt_af,status_b4,status_af,
		dt_created,dt_modified,created_by,modified_by,status)
		 values
	    (o_exist,TODAY,i_audit_action,i_cp_no,i_entity_type,i_entity_no,o_accr_amt,0.00,i_old_status,i_status,
	    current,current,i_modified_by,i_modified_by,'A');
	   
	   if i_audit_action == 'CPV' then 
	   			insert into rms_ricp_cv_log
				(ricp_id,entity_type,entity_no,cp_no,dt_void,dt_cancel,
				dt_created,dt_modified,created_by,modified_by,status)
				 values
			    (o_exist,i_entity_type,i_entity_no,i_cp_no,TODAY,NULL,
			    current,current,i_modified_by,i_modified_by,'A');
			   	update rms_ricp set dt_void=TODAY
			    where ricp_id = o_exist;
	   end if;
		   
	  if i_audit_action == 'CPC' then
	   			insert into rms_ricp_cv_log
				(ricp_id,entity_type,entity_no,cp_no,dt_void,dt_cancel,
				dt_created,dt_modified,created_by,modified_by,status)
				 values
			    (o_exist,i_entity_type,i_entity_no,i_cp_no,NULL,TODAY,
			    current,current,i_modified_by,i_modified_by,'A');	  	
			   	update rms_ricp set dt_cancel=TODAY
			    where ricp_id = o_exist;
	  end if;
	 
	  let o_result = dbinfo('sqlca.sqlerrd2');
	 
	end if;
	
	commit work;
	
	if o_result > 0 then
	    return o_result;
	else
	    return -1;
	end if;

end procedure;
GO
CREATE OR REPLACE PROCEDURE sp_updricpstatus (i_ricp_id integer, i_status nvarchar(10), i_modified_by nvarchar(25))
	returning int as row_count;

	Define row_count int;

	update rms_ricp
	set status=i_status,
	dt_modified = current,
	modified_by=i_modified_by
    where ricp_id = i_ricp_id;

	let row_count = dbinfo('sqlca.sqlerrd2');

	RETURN row_count;

END PROCEDURE;     
GO

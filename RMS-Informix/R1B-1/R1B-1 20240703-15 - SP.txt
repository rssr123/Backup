
CREATE OR REPLACE PROCEDURE sp_checkuserperm(
    i_username NVARCHAR(25),
    i_perm_cd_list VARCHAR(255)
) 
RETURNING LVARCHAR(500) AS o_has_access;

    DEFINE o_has_access LVARCHAR(500);
    DEFINE t_user_role_id INT;

    SELECT sp_group_concat(t4.perm_cd) into o_has_access
    FROM rms_user_role t5
    LEFT OUTER JOIN rms_user t1 ON t1.ssm4uuserrefno = t5.ssm4uuserrefno
    LEFT OUTER JOIN rms_role t2 ON t5.role_id = t2.role_id
    LEFT OUTER JOIN rms_role_perm t3 ON t3.role_id = t2.role_id
    LEFT OUTER JOIN rms_perm t4 ON t3.perm_id = t4.perm_id
    WHERE i_username = t5.ssm4uuserrefno
    AND t4.perm_cd IN (SELECT * FROM TABLE (regex_split(i_perm_cd_list, ','))) AND is_allow = 1;

    RETURN o_has_access;

END PROCEDURE;
GO
CREATE OR REPLACE PROCEDURE sp_generateAgingReport(asOfDate nvarchar(10), rcptNo nvarchar(20), pgTxnId nvarchar(30), stmtNo nvarchar(25),
		matchRec nvarchar(10), pgDtStmtFr date, pgDtStmtTo date, showDupTrans boolean)
	returning datetime year to fraction(5), nvarchar(255), nvarchar(255), nvarchar(255), decimal(16,2),
			datetime year to fraction(5), nvarchar(255), nvarchar(255), nvarchar(255), decimal(16,2),
			integer;
		 
			DEFINE o_dn_dt datetime year to fraction(5);
			DEFINE o_dn_txn_id nvarchar(255);
			DEFINE o_dn_rms_bn nvarchar(255);
			DEFINE o_dn_fms_ref nvarchar(255);
			DEFINE o_dn_amt decimal(16,2);
			DEFINE o_cn_dt datetime year to fraction(5);
			DEFINE o_cn_txn_id nvarchar(255);
			DEFINE o_cn_rms_bn nvarchar(255);
			DEFINE o_cn_fms_ref nvarchar(255);
			DEFINE o_cn_amt decimal(16,2);
			DEFINE o_status integer;
			
			define o_statement lvarchar(2000);
			let o_statement = "select drb.dt_created as dn_dt, pg.pg_txn_id as dn_txn_id, drh.rms_batch_no as dn_rms_bn, drh.fms_ref_no as dn_fms_ref, drh.amt as dn_amt, crb.dt_created as cn_dt, case when (crb.mtt_pg_id is null or crb.mtt_pg_id = '') then null else pg.pg_txn_id end as cn_txn_id, case when (crb.mtt_pg_id is null or crb.mtt_pg_id = '') then null else crh.rms_batch_no end as cn_rms_bn, case when (crb.mtt_pg_id is null or crb.mtt_pg_id = '') then null else crh.fms_ref_no end as cn_fms_ref, case when (crb.mtt_pg_id is null or crb.mtt_pg_id = '') then null else crh.amt end as cn_amt, case when (crb.mtt_pg_id is null or crb.mtt_pg_id = '') then date('" || asOfDate || "') - cast(drb.dt_created as date) else cast(crb.dt_created as date) - cast(drb.dt_created as date) end as status from rms_fms_drmemo_b drb left join rms_fms_crmemo_b crb on crb.mtt_pg_id = drb.mtt_pg_id left join rms_mtt_pg pg on drb.mtt_pg_id = pg.mtt_pg_id left join rms_fms_drmemo_h drh on drb.drmemo_hid = drh.drmemo_hid left join rms_fms_crmemo_h crh on crb.crmemo_hid = crh.crmemo_hid left join rms_mtt_rcpt rcpt on drb.mtt_pg_id = rcpt.mtt_pg_id left join rms_rc_pgmtt rcpgmtt on drb.mtt_pg_id = rcpgmtt.mtt_pg_id left join rms_rc_pg rcpg on rcpgmtt.rc_pg_id = rcpg.rc_pg_id  where drb.dt_created <= date('" || asOfDate || "') and rcpg.dt_settlement <= date('" || asOfDate || "')";
			if (rcptNo is not null and rcptNo != '') then
				let o_statement = o_statement || " and rcpt.rcpt_no like '%" || rcptNo || "'";
			end if
			if (pgTxnId is not null and pgTxnId != '') then
				let o_statement = o_statement || " and pg.pg_txn_id like '%" || pgTxnId || "'";
			end if
			if (stmtNo is not null and stmtNo != '') then
				let o_statement = o_statement || " and rcpg.stmt_no like '%" || stmtNo || "'";
			end if
			if (pgDtStmtFr is not null and pgDtStmtTo is not null) then
				let o_statement = o_statement || " and cast(rcpg.dt_settlement as date) >= " || pgDtStmtFr || " and cast(rcpg.dt_settlement as date) <= " || pgDtStmtTo;
			end if
			if lower(matchRec) = 'matched' then
				let o_statement = o_statement || " and crb.dt_created is not null";
			elif lower(matchRec) = 'unmatched' then
				let o_statement = o_statement || " and crb.dt_created is null";
			end if
			if (showDupTrans = 't') then
				let o_statement = o_statement || " and (rcpt.rcpt_no is null or rcpt.rcpt_no = '')";
			end if
			let o_statement = o_statement || ";";
			
			prepare s from o_statement;
			declare c cursor for s;
			open c;
			while(1 = 1)
				fetch c into o_dn_dt, o_dn_txn_id, o_dn_rms_bn, o_dn_fms_ref, o_dn_amt, o_cn_dt, o_cn_txn_id, o_cn_rms_bn, o_cn_fms_ref, o_cn_amt, o_status;
				IF (SQLCODE != 100) THEN
			    	RETURN o_dn_dt, o_dn_txn_id, o_dn_rms_bn, o_dn_fms_ref, o_dn_amt, o_cn_dt, o_cn_txn_id, o_cn_rms_bn, o_cn_fms_ref, o_cn_amt, o_status WITH RESUME;
			  	ELSE
			    	EXIT;
			  	END IF
			END while
			
			close c;
			free c;
			free s;
end procedure
GO
CREATE OR REPLACE PROCEDURE sp_generatericpagingreport(asOfDate nvarchar(10), issDtFr date, issDtTo date, showExp boolean, showCanVoid boolean, 
								entityType nvarchar(50), entityNm nvarchar(100), pymtDtFr date, pymtDtTo date,
								expDtFr date, expDtTo date, woDtFr date, woDtTo date, canDtFr date, canDtTo date,
								voidDtFr date, voidDtTo date)
	returning nvarchar(30), nvarchar(25), nvarchar(50), nvarchar(100), nvarchar(50), decimal(16,2),
			decimal(16,2), date, datetime year to fraction(5), date, date, date, date, integer;
		 
			DEFINE o_bat_no nvarchar(30);
			DEFINE o_fms_ref_no nvarchar(25);
			DEFINE o_entity_type nvarchar(50);
			DEFINE o_entity_nm nvarchar(100);
			DEFINE o_ricp_status nvarchar(50);
			DEFINE o_cp_amt decimal(16,2);
			DEFINE o_net_amt decimal(16,2);
			DEFINE o_dt_issuance date;
			DEFINE o_pymt_dt datetime year to fraction(5);
			DEFINE o_dt_expiry date;
			DEFINE o_dt_wo date;
			DEFINE o_dt_can date;
			DEFINE o_dt_void date;
			DEFINE o_status integer;
					
			define o_statement lvarchar(2000);
			let o_statement = "select fjh.batch_no, fjh.fms_ref_no, fullEntT.nm_en, itm.entity_nm, fullStatusName.nm_en, ricp.cp_amt, itm.net_amt, ricp.dt_issuance, rcp.rcpt_dt, ricp.dt_expiry, ricp.dt_writeoff, ricp.dt_cancel, ricp.dt_void, date('" || asOfDate || "') - ricp.dt_issuance as status from rms_ricp ricp left join (select SUBSTR(SUBSTR(attr_ext_ref_no, CHARINDEX('-', attr_ext_ref_no)+1), 1, CHARINDEX('-', SUBSTR(attr_ext_ref_no, CHARINDEX('-', attr_ext_ref_no)))) as ricp_id, batch_no, fms_ref_no from rms_fms_jn_h) fjh on ricp.ricp_id = fjh.ricp_id left join rms_mtt_item itm on ricp.cp_no = itm.cp_no left join rms_mtt_rcpt rcp on itm.mtt_id = rcp.mtt_id left join (select param_cd, nm_en from rms_param where param_grp_nm = 'RICP-Status') as fullStatusName on ricp.status = fullStatusName.param_cd left join (select param_cd, nm_en from rms_param where param_grp_nm = 'EntityType') as fullEntT on ricp.entity_type = fullEntT.param_cd where ricp.dt_issuance <= date('" || asOfDate || "') and ricp.dt_issuance >= date('" || issDtFr || "') and ricp.dt_issuance <= date('" || issDtTo || "')";

			if (showExp = 'f') then
				let o_statement = o_statement || " and ricp.status != 'WO'";
			end if
			
			if (showCanVoid = 'f') then
				let o_statement = o_statement || " and ricp.status != 'C' and ricp.status != 'V'";
			end if
			
			if (entityType is not null and entityType != '') then
				let o_statement = o_statement || " and ricp.entityType = '" || entityType || "'";
			end if
			
			if (entityNm is not null and entityNm != '') then
				let o_statement = o_statement || " and itm.entity_nm like '%" || entityNm || "'";
			end if
			
			if (pymtDtFr is not null and pymtDtTo is not null) then
				let o_statement = o_statement || " and cast(rcp.rcpt_dt as date) >= " || pymtDtFr || " and cast(rcp.rcpt_dt as date) <= " || pymtDtTo;
			end if
			
			if (expDtFr is not null and expDtTo is not null) then
				let o_statement = o_statement || " and ricp.dt_expiry >= " || expDtFr || " and ricp.dt_expiry <= " || expDtTo;
			end if
			
			if (woDtFr is not null and woDtTo is not null) then
				let o_statement = o_statement || " and ricp.dt_writeoff >= " || woDtFr || " and ricp.dt_writeoff <= " || woDtTo;
			end if
			
			if (canDtFr is not null and canDtTo is not null) then
				let o_statement = o_statement || " and ricp.dt_cancel >= " || canDtFr || " and ricp.dt_cancel <= " || canDtTo;
			end if
			
			if (voidDtFr is not null and voidDtTo is not null) then
				let o_statement = o_statement || " and ricp.dt_void >= " || voidDtFr || " and ricp.dt_void <= " || voidDtTo;
			end if
			
			let o_statement = o_statement || ";";
			
			prepare s from o_statement;
			declare c cursor for s;
			open c;
			while(1 = 1)
				fetch c into o_bat_no,o_fms_ref_no,o_entity_type,o_entity_nm,o_ricp_status,o_cp_amt,o_net_amt,o_dt_issuance,o_pymt_dt,o_dt_expiry,o_dt_wo,o_dt_can,o_dt_void,o_status;
				IF (SQLCODE != 100) THEN
			    	RETURN o_bat_no,o_fms_ref_no,o_entity_type,o_entity_nm,o_ricp_status,o_cp_amt,o_net_amt,o_dt_issuance,o_pymt_dt,o_dt_expiry,o_dt_wo,o_dt_can,o_dt_void,o_status WITH RESUME;
			  	ELSE
			    	EXIT;
			  	END IF
			END while
			
			close c;
			free c;
			free s;
end procedure
GO

create or replace procedure sp_getfmsjn()
returning nvarchar(30) as branch_id, lvarchar(256) as desc,  nvarchar(10) as ledger_id,
nvarchar(2) as module, DATE as dt_txn, lvarchar(256) as attr_ext_ref_no,lvarchar(256) as attr_ext_sys,
nvarchar(10) as acct1, nvarchar(30) as branch1, decimal(16,2) as credit1, decimal(16,2) as debit1, nvarchar(30) as sub_acct1, lvarchar(256) as txn_desc1,
nvarchar(10) as acct2, nvarchar(30) as branch2, decimal(16,2) as credit2, decimal(16,2) as debit2, nvarchar(30) as sub_acct2, lvarchar(256) as txn_desc2;

define o_branch_id nvarchar(30); 
define o_desc lvarchar(256);
define o_ledger_id nvarchar(10);
define o_module nvarchar(2);
define o_dt_txn DATE;
define o_attr_ext_ref_no lvarchar(256);
define o_attr_ext_sys lvarchar(256);
define o_acct1 nvarchar(10);
define o_branch1 nvarchar(30);
define o_credit1 decimal(16,2);
define o_debit1  decimal(16,2);
define o_sub_acct1  nvarchar(30);
define o_txn_desc1  lvarchar(256);
define o_acct2 nvarchar(10);
define o_branch2 nvarchar(30);
define o_credit2 decimal(16,2);
define o_debit2 decimal(16,2);
define o_sub_acct2 nvarchar(30);
define o_txn_desc2 lvarchar(256);

foreach c_getfmsjn for     
    select 
        t1.branch_id, t1.desc, t1.ledger_id, t1.module, t1.dt_txn, t1.attr_ext_ref_no, t1.attr_ext_sys,
        max(case when row_num = 1 then acct end) as acct1,
        max(case when row_num = 1 then branch end) as branch1,
        max(case when row_num = 1 then credit end) as credit1,
        max(case when row_num = 1 then debit end) as debit1,
        max(case when row_num = 1 then sub_acct end) as sub_acct1,
        max(case when row_num = 1 then txn_desc end) as txn_desc1,
        max(case when row_num = 2 then acct end) as acct2,
        max(case when row_num = 2 then branch end) as branch2,
        max(case when row_num = 2 then credit end) as credit2,
        max(case when row_num = 2 then debit end) as debit2,
        max(case when row_num = 2 then sub_acct end) as sub_acct2,
        max(case when row_num = 2 then txn_desc end) as txn_desc2
    into o_branch_id, o_desc, o_ledger_id, o_module, o_dt_txn, o_attr_ext_ref_no, o_attr_ext_sys, 
         o_acct1, o_branch1, o_credit1, o_debit1, o_sub_acct1, o_txn_desc1,
         o_acct2, o_branch2, o_credit2, o_debit2, o_sub_acct2, o_txn_desc2
    from rms_fms_jn_h t1
    join (
        select 
            jn_hid, acct, branch, credit, debit, sub_acct, txn_desc,
            row_number() over (partition by jn_hid order by acct) as row_num
        from rms_fms_jn_b
    ) as t2 on t1.jn_hid = t2.jn_hid
    where t1.fms_ref_no is null

    -- get limit per payment
    select config_value
    into ari_limit
    from rms_config
    where config_cd = 'FMS_ARI_Details';

    -- filter in rms_fms_arrb's mtt record
    -- get the most earliest date
    select min(date(pg_resp_time)) as resp_time
    into temp_date
    from rms_mtt_pg
    where (in_fms_posting = 0 or in_fms_posting is null)
    and pg_txn_status=0
    and date(pg_resp_time) <= today-1;

    -- customer field
    select acct_cd
    into o_customer
    from rms_fms_acct
    where acct_nm ="AR Invoice's RMS PG MID Account ID";

       t2.item_desc,

          AND (in_fms_posting = 0 OR in_fms_posting IS NULL)
          AND pg_txn_status = 0
        ORDER BY pg_resp_time
    ) t1
    INNER JOIN rms_mtt_item t2 ON t1.mtt_id = t2.mtt_id
    INNER JOIN rms_mtt t3 ON t1.mtt_id = t3.mtt_id

    INT8 AS rpt_ripl_age_id,
    DATETIME YEAR TO SECOND AS p_dt_req,
    INT AS p_imp_status,
    INT AS p_exp_status,
    NVARCHAR(1) AS p_ent_ty,
    NVARCHAR(100) AS p_ent_nm,
    DATETIME YEAR TO SECOND AS p_dt_due_fr,
    DATETIME YEAR TO SECOND AS p_dt_due_to,
    DATETIME YEAR TO SECOND AS p_dt_rcpt_fr,
    DATETIME YEAR TO SECOND AS p_dt_rcpt_to,
    DATETIME YEAR TO SECOND AS p_dt_imp_fr,
    DATETIME YEAR TO SECOND AS p_dt_imp_to,
    DATETIME YEAR TO SECOND AS p_dt_wo_fr,
    DATETIME YEAR TO SECOND AS p_dt_wo_to,
    DATETIME YEAR TO SECOND AS dt_created,
    DATETIME YEAR TO SECOND AS dt_modified,
    NVARCHAR(25) AS created_by,
    NVARCHAR(25) AS modified_by,
    NVARCHAR(10) AS status,
    nvarchar(50) as p_email,
	nvarchar(50) as p_file_type,
	int as p_file_size,
	nvarchar(255) as p_file_nm,
	nvarchar(30) as p_batch_no,
	nvarchar(25) as p_fms_ref_no,
	nvarchar(10) as task_id;

DEFINE o_rpt_ripl_age_id INT8;
DEFINE o_p_dt_req DATETIME YEAR TO SECOND;
DEFINE o_p_imp_status INT;
DEFINE o_p_exp_status INT;
DEFINE o_p_ent_ty NVARCHAR(1);
DEFINE o_p_ent_nm NVARCHAR(100);
DEFINE o_p_dt_due_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_due_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_rcpt_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_rcpt_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_imp_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_imp_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_wo_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_wo_to DATETIME YEAR TO SECOND;
DEFINE o_dt_created DATETIME YEAR TO SECOND;
DEFINE o_dt_modified DATETIME YEAR TO SECOND;
DEFINE o_created_by NVARCHAR(25);
DEFINE o_modified_by NVARCHAR(25);
DEFINE o_status NVARCHAR(10);
define o_p_email nvarchar(50);
define o_p_file_type nvarchar(50);
define o_p_file_size int;
define o_p_file_nm nvarchar(255);
define o_p_batch_no	nvarchar(30);
define o_p_fms_ref_no nvarchar(25);




FOREACH c_getriplaging FOR
    select t1.rpt_ripl_age_id, t1.p_dt_req, t1.p_imp_status, t1.p_exp_status, t1.p_ent_ty, t1.p_ent_nm,
           t1.p_dt_due_fr, t1.p_dt_due_to, t1.p_dt_rcpt_fr, t1.p_dt_rcpt_to, t1.p_dt_imp_fr, t1.p_dt_imp_to,
           t1.p_dt_wo_fr, t1.p_dt_wo_to, t1.dt_created, t1.dt_modified, t1.created_by, t1.modified_by,
           t1.status, t1.p_email, t1.p_file_type, t1.p_file_size, t1.p_file_nm, 
	       t1.p_batch_no, t1.p_fms_ref_no
    INTO o_rpt_ripl_age_id, o_p_dt_req, o_p_imp_status, o_p_exp_status, o_p_ent_ty, o_p_ent_nm,
         o_p_dt_due_fr, o_p_dt_due_to, o_p_dt_rcpt_fr, o_p_dt_rcpt_to, o_p_dt_imp_fr, o_p_dt_imp_to,
         o_p_dt_wo_fr, o_p_dt_wo_to, o_dt_created, o_dt_modified, o_created_by, o_modified_by,
         o_status, o_p_email, o_p_file_type, o_p_file_size, o_p_file_nm,
	    o_p_batch_no, o_p_fms_ref_no
    FROM rms_rpt_ripl_age t1
    
    where t1.status = 'P' or t1.status = 'I'
      

    RETURN o_rpt_ripl_age_id, o_p_dt_req, o_p_imp_status, o_p_exp_status, o_p_ent_ty, o_p_ent_nm,
           o_p_dt_due_fr, o_p_dt_due_to, o_p_dt_rcpt_fr, o_p_dt_rcpt_to, o_p_dt_imp_fr, o_p_dt_imp_to,
           o_p_dt_wo_fr, o_p_dt_wo_to, o_dt_created, o_dt_modified, o_created_by, o_modified_by,
           o_status, o_p_email, o_p_file_type, o_p_file_size, o_p_file_nm, 
	       o_p_batch_no, o_p_fms_ref_no, 'RG'||lpad(o_rpt_ripl_age_id,6,'0') with resume;
END FOREACH


	
END PROCEDURE
GO

CREATE OR REPLACE PROCEDURE sp_getriplaging (
	i_page int, 
	i_size int,
    i_rpt_ripl_age_id INT8,
    i_p_dt_req DATE,
    i_p_imp_status INT,
    i_p_exp_status INT,
    i_p_ent_ty NVARCHAR(1),
    i_p_ent_nm NVARCHAR(200),
    i_p_dt_due_fr DATE,
    i_p_dt_due_to DATE,
    i_p_dt_rcpt_fr DATE,
    i_p_dt_rcpt_to DATE,
    i_p_dt_imp_fr DATE,
    i_p_dt_imp_to DATE,
    i_p_dt_wo_fr DATE,
    i_p_dt_wo_to DATE,
    i_created_by NVARCHAR(25),
    i_modified_by NVARCHAR(25),
    i_status NVARCHAR(10),
    i_p_email NVARCHAR(255),
    i_p_file_type NVARCHAR(5)
)
RETURNING
    INT8 AS rpt_ripl_age_id,
    DATETIME YEAR TO SECOND AS p_dt_req,
    INT AS p_imp_status,
    INT AS p_exp_status,
    NVARCHAR(1) AS p_ent_ty,
    NVARCHAR(100) AS p_ent_nm,
    DATETIME YEAR TO SECOND AS p_dt_due_fr,
    DATETIME YEAR TO SECOND AS p_dt_due_to,
    DATETIME YEAR TO SECOND AS p_dt_rcpt_fr,
    DATETIME YEAR TO SECOND AS p_dt_rcpt_to,
    DATETIME YEAR TO SECOND AS p_dt_imp_fr,
    DATETIME YEAR TO SECOND AS p_dt_imp_to,
    DATETIME YEAR TO SECOND AS p_dt_wo_fr,
    DATETIME YEAR TO SECOND AS p_dt_wo_to,
    DATETIME YEAR TO SECOND AS dt_created,
    DATETIME YEAR TO SECOND AS dt_modified,
    NVARCHAR(25) AS created_by,
    NVARCHAR(25) AS modified_by,
    NVARCHAR(10) AS status,
    NVARCHAR(50) AS p_email,
    NVARCHAR(50) AS p_file_type,
    NVARCHAR(10) AS task_id,
    INT AS total;

DEFINE o_rpt_ripl_age_id INT8;
DEFINE o_p_dt_req DATETIME YEAR TO SECOND;
DEFINE o_p_imp_status INT;
DEFINE o_p_exp_status INT;
DEFINE o_p_ent_ty NVARCHAR(1);
DEFINE o_p_ent_nm NVARCHAR(100);
DEFINE o_p_dt_due_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_due_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_rcpt_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_rcpt_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_imp_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_imp_to DATETIME YEAR TO SECOND;
DEFINE o_p_dt_wo_fr DATETIME YEAR TO SECOND;
DEFINE o_p_dt_wo_to DATETIME YEAR TO SECOND;
DEFINE o_dt_created DATETIME YEAR TO SECOND;
DEFINE o_dt_modified DATETIME YEAR TO SECOND;
DEFINE o_created_by NVARCHAR(25);
DEFINE o_modified_by NVARCHAR(25);
DEFINE o_status NVARCHAR(10);
DEFINE o_p_email NVARCHAR(50);
DEFINE o_p_file_type NVARCHAR(50);
define calculatedpage int;
DEFINE total INT;

let calculatedpage = (i_page-1)*i_size;

LET total = (SELECT COUNT(1)
             FROM rms_rpt_ripl_age t1
             WHERE (i_rpt_ripl_age_id IS NULL OR t1.rpt_ripl_age_id = i_rpt_ripl_age_id)
               AND (i_p_dt_req IS NULL OR t1.p_dt_req = i_p_dt_req)
               AND (i_p_imp_status IS NULL OR t1.p_imp_status = i_p_imp_status)
               AND (i_p_exp_status IS NULL OR t1.p_exp_status = i_p_exp_status)
               AND (i_p_ent_ty IS NULL OR t1.p_ent_ty like '%' || i_p_ent_nm ||'%')
               AND (i_p_ent_nm IS NULL OR t1.p_ent_nm = i_p_ent_nm)
               AND (i_p_dt_due_fr IS NULL OR t1.p_dt_due_fr >= i_p_dt_due_fr)
               AND (i_p_dt_due_to IS NULL OR t1.p_dt_due_to <= i_p_dt_due_to)
               AND (i_p_dt_rcpt_fr IS NULL OR t1.p_dt_rcpt_fr >= i_p_dt_rcpt_fr)
               AND (i_p_dt_rcpt_to IS NULL OR t1.p_dt_rcpt_to <= i_p_dt_rcpt_to)
               AND (i_p_dt_imp_fr IS NULL OR t1.p_dt_imp_fr >= i_p_dt_imp_fr)
               AND (i_p_dt_imp_to IS NULL OR t1.p_dt_imp_to <= i_p_dt_imp_to)
               AND (i_p_dt_wo_fr IS NULL OR t1.p_dt_wo_fr >= i_p_dt_wo_fr)
               AND (i_p_dt_wo_to IS NULL OR t1.p_dt_wo_to <= i_p_dt_wo_to)
               AND (i_created_by IS NULL OR t1.created_by = i_created_by)
               AND (i_modified_by IS NULL OR t1.modified_by = i_modified_by)
               AND (i_status IS NULL OR t1.status = i_status)
               AND (i_p_email IS NULL OR t1.p_email = i_p_email)
               AND (i_p_file_type IS NULL OR t1.p_file_type = i_p_file_type));
			   
if i_size is not null then  

FOREACH c_getriplaging FOR
    select skip calculatedpage limit i_size t1.rpt_ripl_age_id, t1.p_dt_req, t1.p_imp_status, t1.p_exp_status, t1.p_ent_ty, t1.p_ent_nm,
           t1.p_dt_due_fr, t1.p_dt_due_to, t1.p_dt_rcpt_fr, t1.p_dt_rcpt_to, t1.p_dt_imp_fr, t1.p_dt_imp_to,
           t1.p_dt_wo_fr, t1.p_dt_wo_to, t1.dt_created, t1.dt_modified, t1.created_by, t1.modified_by,
           t1.status, t1.p_email, t1.p_file_type
    INTO o_rpt_ripl_age_id, o_p_dt_req, o_p_imp_status, o_p_exp_status, o_p_ent_ty, o_p_ent_nm,
         o_p_dt_due_fr, o_p_dt_due_to, o_p_dt_rcpt_fr, o_p_dt_rcpt_to, o_p_dt_imp_fr, o_p_dt_imp_to,
         o_p_dt_wo_fr, o_p_dt_wo_to, o_dt_created, o_dt_modified, o_created_by, o_modified_by,
         o_status, o_p_email, o_p_file_type
    FROM rms_rpt_ripl_age t1
    WHERE (i_rpt_ripl_age_id IS NULL OR t1.rpt_ripl_age_id = i_rpt_ripl_age_id)
      AND (i_p_dt_req IS NULL OR t1.p_dt_req <= i_p_dt_req)
      AND (i_p_imp_status IS NULL OR t1.p_imp_status = i_p_imp_status)
      AND (i_p_exp_status IS NULL OR t1.p_exp_status = i_p_exp_status)
      AND (i_p_ent_ty IS NULL OR t1.p_ent_ty = i_p_ent_ty)
      AND (i_p_ent_nm IS NULL OR t1.p_ent_nm = i_p_ent_nm)
      AND (i_p_dt_due_fr IS NULL OR t1.p_dt_due_fr = i_p_dt_due_fr)
      AND (i_p_dt_due_to IS NULL OR t1.p_dt_due_to = i_p_dt_due_to)
      AND (i_p_dt_rcpt_fr IS NULL OR t1.p_dt_rcpt_fr = i_p_dt_rcpt_fr)
      AND (i_p_dt_rcpt_to IS NULL OR t1.p_dt_rcpt_to = i_p_dt_rcpt_to)
      AND (i_p_dt_imp_fr IS NULL OR t1.p_dt_imp_fr = i_p_dt_imp_fr)
      AND (i_p_dt_imp_to IS NULL OR t1.p_dt_imp_to = i_p_dt_imp_to)
      AND (i_p_dt_wo_fr IS NULL OR t1.p_dt_wo_fr = i_p_dt_wo_fr)
      AND (i_p_dt_wo_to IS NULL OR t1.p_dt_wo_to = i_p_dt_wo_to)
      AND (i_created_by IS NULL OR t1.created_by = i_created_by)
      AND (i_modified_by IS NULL OR t1.modified_by = i_modified_by)
      AND (i_status IS NULL OR t1.status = i_status)
      AND (i_p_email IS NULL OR t1.p_email = i_p_email)
      AND (i_p_file_type IS NULL OR t1.p_file_type = i_p_file_type)
    ORDER BY t1.dt_modified DESC

    RETURN o_rpt_ripl_age_id, o_p_dt_req, o_p_imp_status, o_p_exp_status, o_p_ent_ty, o_p_ent_nm,
           o_p_dt_due_fr, o_p_dt_due_to, o_p_dt_rcpt_fr, o_p_dt_rcpt_to, o_p_dt_imp_fr, o_p_dt_imp_to,
           o_p_dt_wo_fr, o_p_dt_wo_to, o_dt_created, o_dt_modified, o_created_by, o_modified_by,
           o_status, o_p_email, o_p_file_type, 'RG'||lpad(o_rpt_ripl_age_id,6,'0'), total WITH RESUME;
END FOREACH

ELSE
foreach c_getriplaging for
	SELECT t1.rpt_ripl_age_id, t1.p_dt_req, t1.p_imp_status, t1.p_exp_status, t1.p_ent_ty, t1.p_ent_nm,

CREATE OR REPLACE PROCEDURE sp_getutldays(i_period_key DATE)
RETURNING 
    nvarchar(20) AS period,
    DECIMAL(15,5) AS in,
    DECIMAL(15,5) AS out,
    DECIMAL(15,5) AS variance,
    nvarchar(10) AS periodbalance,
    DATE AS dummydate;

DEFINE o_period nvarchar(20);
DEFINE o_in DECIMAL(15,5);
DEFINE o_out DECIMAL(15,5);
DEFINE o_variance DECIMAL(15,5);
DEFINE o_period_balance nvarchar(10);
DEFINE o_dummydate DATE;
DEFINE total_in DECIMAL(15,5);
DEFINE total_out DECIMAL(15,5);
DEFINE total_variance DECIMAL(15,5);
DEFINE total_period_balance DECIMAL(15,5);

-- Initialize totals
LET total_in = 0;
LET total_out = 0;
LET total_variance = 0;
LET total_period_balance = 0;

-- Cursor to fetch data
FOREACH cursor1 FOR

    WITH days_of_month(day_key) AS (
    SELECT MDY(MONTH(i_period_key), 1, YEAR(i_period_key)) -- First day of the input month
    UNION ALL
    SELECT day_key + 1
    FROM days_of_month
    WHERE day_key < LAST_DAY(i_period_key)
    )
    SELECT 
        CASE 
            WHEN d.day_key > TODAY THEN '' 
            ELSE TO_CHAR(d.day_key, '%d %b %Y')
        END AS period,
        CASE 
            WHEN d.day_key > TODAY THEN NULL 
            ELSE NVL(SUM(NVL(r.debit_amt, 0)), 0) 
        END AS in,
        CASE 
            WHEN d.day_key > TODAY THEN NULL 
            ELSE NVL(SUM(NVL(r.credit_amt, 0)), 0) 
        END AS out,
        CASE 
            WHEN d.day_key > TODAY THEN NULL 
            ELSE NVL(SUM(NVL(r.variance, 0)), 0) 
        END AS variance,
        CASE 
            WHEN d.day_key > TODAY THEN '' 
            ELSE TO_CHAR(ROUND(NVL(SUM(NVL(r.accumulate_amt, 0)), 0), 2))
        END AS periodbalance,
        d.day_key AS dummydate
    INTO
        o_period,
        o_in,
        o_out,
        o_variance,
        o_period_balance,
        o_dummydate
    FROM
        days_of_month d
    LEFT JOIN
        rms_unmatch_txn r ON d.day_key = r.period_key
    GROUP BY
        d.day_key
    ORDER BY
        d.day_key
	
    -- Accumulate totals only if the date is not in the future
    IF o_dummydate <= today THEN
        LET total_in = total_in + NVL(o_in, 0);
        LET total_out = total_out + NVL(o_out, 0);
        LET total_variance = total_variance + NVL(o_variance, 0);
        LET total_period_balance = total_period_balance + NVL(CAST(o_period_balance AS DECIMAL(15,5)), 0);
    END IF;

    RETURN o_period, o_in, o_out, o_variance, o_period_balance, o_dummydate WITH RESUME;
END FOREACH;

-- Add totals as the last row
RETURN 'Total', total_in, total_out, total_variance, 'N/A', NULL;
END PROCEDURE;
GO
CREATE OR REPLACE PROCEDURE sp_getutlmonths(i_period_key DATE)
RETURNING 
    nvarchar(20) AS period,
    DECIMAL(15,5) AS in,
    DECIMAL(15,5) AS out,
    DECIMAL(15,5) AS variance,
    nvarchar(10) AS periodbalance,
    DATE AS dummydate;

DEFINE o_period nvarchar(20);
DEFINE o_in DECIMAL(15,5);
DEFINE o_out DECIMAL(15,5);
DEFINE o_variance DECIMAL(15,5);
DEFINE o_period_balance nvarchar(10);
DEFINE o_dummydate DATE;
DEFINE o_month_key INTEGER;
DEFINE total_previous_year_balance DECIMAL(15,5);

-- Calculate the total amount for the period balance of the previous year
SELECT COALESCE(SUM(accumulate_amt), 0) 
INTO total_previous_year_balance 
FROM rms_unmatch_txn 
WHERE YEAR(period_key) = YEAR(i_period_key) - 1;

-- Add "Beginning Balance" row at the top
RETURN 'Beginning Balance', 0.00, 0.00, 0.00, TO_CHAR(ROUND(total_previous_year_balance, 2)), NULL::DATE WITH RESUME;

FOREACH cursor1 FOR
    SELECT 
        CASE 
            WHEN DummyDate > TODAY THEN '' 
            ELSE Period 
        END AS Period,
        CASE 
            WHEN DummyDate > TODAY THEN NULL 
            ELSE COALESCE(SUM(In), 0) 
        END AS In,
        CASE 
            WHEN DummyDate > TODAY THEN NULL 
            ELSE COALESCE(SUM(Out), 0) 
        END AS Out,
        CASE 
            WHEN DummyDate > TODAY THEN NULL 
            ELSE COALESCE(SUM(Variance), 0) 
        END AS Variance,
        CASE 
            WHEN MonthKey = 13 THEN 'N/A' 
            WHEN DummyDate > TODAY THEN '' 
            ELSE TO_CHAR(ROUND(COALESCE(SUM(PeriodBalance), 0), 2))
        END AS PeriodBalance,
        DummyDate,
        MonthKey
    INTO
        o_period,
        o_in,
        o_out,
        o_variance,
        o_period_balance,
        o_dummydate,
        o_month_key
    FROM (
        SELECT
            CASE MONTH(period_key)
                WHEN 1 THEN 'January'
                WHEN 2 THEN 'February'
                WHEN 3 THEN 'March'
                WHEN 4 THEN 'April'
                WHEN 5 THEN 'May'
                WHEN 6 THEN 'June'
                WHEN 7 THEN 'July'
                WHEN 8 THEN 'August'
                WHEN 9 THEN 'September'
                WHEN 10 THEN 'October'
                WHEN 11 THEN 'November'
                WHEN 12 THEN 'December'
            END AS Period,
            debit_amt AS In, 
            credit_amt AS Out, 
            variance AS Variance, 
            accumulate_amt AS PeriodBalance,
            MDY(MONTH(period_key), 1, YEAR(i_period_key)) AS DummyDate,
            MONTH(period_key) AS MonthKey
        FROM 
            rms_unmatch_txn
        WHERE 
            YEAR(period_key) = YEAR(i_period_key)
        UNION ALL
        SELECT
            CASE n
                WHEN 1 THEN 'January'
                WHEN 2 THEN 'February'
                WHEN 3 THEN 'March'
                WHEN 4 THEN 'April'
                WHEN 5 THEN 'May'
                WHEN 6 THEN 'June'
                WHEN 7 THEN 'July'
                WHEN 8 THEN 'August'
                WHEN 9 THEN 'September'
                WHEN 10 THEN 'October'
                WHEN 11 THEN 'November'
                WHEN 12 THEN 'December'
            END AS Period,
            0 AS In, 
            0 AS Out, 
            0 AS Variance, 
            0 AS PeriodBalance,
            MDY(n, 1, YEAR(i_period_key)) AS DummyDate,
            n AS MonthKey
        FROM (SELECT 1 AS n UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11 UNION ALL SELECT 12) AS Months
    ) AS Subquery
    GROUP BY 
        Period,
        DummyDate,
        MonthKey

    UNION ALL

    SELECT
        'Total' AS Period,
        COALESCE(SUM(In), 0) AS In,
        COALESCE(SUM(Out), 0) AS Out,
        COALESCE(SUM(Variance), 0) AS Variance,
        'N/A' AS PeriodBalance,
        NULL::DATE AS DummyDate,
        13 AS MonthKey
    FROM (
        SELECT
            debit_amt AS In, 
            credit_amt AS Out, 
            variance AS Variance, 
            accumulate_amt AS PeriodBalance,
            0 AS MonthKey
        FROM 
            rms_unmatch_txn
        WHERE 
            YEAR(period_key) = YEAR(i_period_key)
    ) AS TotalSubquery

    ORDER BY 
        MonthKey ASC
    RETURN o_period, o_in, o_out, o_variance, o_period_balance, o_dummydate WITH RESUME;
END FOREACH;

END PROCEDURE;
GO

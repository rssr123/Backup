DROP PROCEDURE PROCEDURE sp_delrilt

GO

CREATE OR REPLACE PROCEDURE sp_delrilt(
    i_lit_no NVARCHAR(20),
    i_lit_item_ref NVARCHAR(100)
    )

    RETURNING INT8 AS result;

    DEFINE o_result INT8;
    DEFINE x_rilt_id INT8;
    DEFINE x_lit_amt_bf DECIMAL(16,2);

    ON EXCEPTION
        ROLLBACK WORK;
        RETURN -4;
    END EXCEPTION;

    BEGIN WORK;

        SET LOCK MODE TO WAIT 30;

        LET o_result = 0;

        IF i_lit_no IS NOT NULL AND i_lit_item_ref IS NOT NULL THEN
            LET o_result = -1; -- InvalidFormat

        --Logic for submitRILTCancel
        ELIF i_lit_no IS NOT NULL THEN

            SELECT count(1) 
            INTO o_result
            FROM rms_rilt
            WHERE lit_no = i_lit_no
            AND status = 'CA';

            IF o_result > 0 THEN

                FOREACH cursor1 FOR
                    SELECT T1.rilt_id AS rilt_id, T1.lit_amt AS lit_amount
                    INTO x_rilt_id, x_lit_amt_bf
                    FROM rms_rilt T1
                    --WHERE (i_lit_no IS NULL OR T1.lit_no LIKE '%' || i_lit_no || '%')
                    WHERE T1.lit_no = i_lit_no
                    AND T1.Status = 'CA'

                    UPDATE rms_rilt
                    SET status = 'C',
                        --lit_amt = 0,
                        dt_modified = current
                    WHERE rilt_id = x_rilt_id;

                    LET o_result = DBINFO('sqlca.sqlerrd2');

                    INSERT INTO rms_rilt_a
                    (
                        rilt_id, lit_amt_bf, lit_amt_af, dt_txn, dt_created, dt_modified, created_by, modified_by, status
                    )
                    VALUES
                    (
                        x_rilt_id, x_lit_amt_bf, 0, current, current, current, 'SYSTEM', 'SYSTEM', 'C'
                    );

                    LET o_result = DBINFO('sqlca.sqlerrd1');

                END FOREACH;
            ELSE
                LET o_result = -2;
            END IF;

        ELIF i_lit_item_ref IS NOT NULL THEN

            IF EXISTS (SELECT 1 FROM rms_rilt 
                        WHERE lit_item_ref = i_lit_item_ref 
                        and status = 'CA') THEN
                SELECT
                    rilt_id, lit_amt
                INTO
                    x_rilt_id, x_lit_amt_bf
                FROM
                    rms_rilt
                WHERE
                    lit_item_ref = i_lit_item_ref
                    AND status = 'CA';
                LET o_result = 1;
            ELSE
                LET o_result = -3; -- System Rule Violation
            END IF;

            IF o_result > 0 THEN
                UPDATE
                    rms_rilt
                SET
                    status = 'C',
                    --lit_amt = 0,
                    dt_modified = current
                WHERE
                    rilt_id = x_rilt_id;
                LET o_result = DBINFO('sqlca.sqlerrd2');
            END IF;

            IF o_result > 0 THEN
                INSERT INTO rms_rilt_a
                    (rilt_id, lit_amt_bf, lit_amt_af, dt_txn, dt_created, dt_modified, created_by, modified_by, status)
                VALUES
                    (x_rilt_id, x_lit_amt_bf, 0, current, current, current, 'SYSTEM', 'SYSTEM', 'C');

                LET o_result = DBINFO('sqlca.sqlerrd1');
            END IF;

        ELSE
            LET o_result = -1; -- InvalidFormat
        END IF;

    IF o_result > 0 THEN
        LET o_result = 1; -- SuccessResponseExternal
        COMMIT WORK;
    ELSE
        ROLLBACK WORK;
    END IF;

RETURN o_result;

END PROCEDURE;

GO 

DROP PROCEDURE sp_insrilt

GO

CREATE OR REPLACE PROCEDURE sp_insrilt(
    i_lit_no NVARCHAR(20),
    i_lit_item_ref NVARCHAR(100),
    i_lit_amount DECIMAL(16,2),
    i_entity_type NVARCHAR(1),
    i_entity_no NVARCHAR(40),
    i_dt_due NVARCHAR(100)
    )

    RETURNING INT8 AS result;

    DEFINE o_result INT8;

    ON EXCEPTION
        ROLLBACK WORK;
    END EXCEPTION;

    BEGIN WORK;

        SET LOCK MODE TO WAIT 30;

        LET o_result = 0;

        IF 
            ((i_lit_no IS NOT NULL) AND i_lit_no != '')
            AND ((i_lit_item_ref IS NOT NULL) AND i_lit_item_ref != '')
            AND ((i_lit_amount IS NOT NULL) AND (i_lit_amount > 0))
            AND ((i_entity_type IS NOT NULL) AND i_entity_type != '')
            AND ((i_entity_no IS NOT NULL) AND i_entity_no != '')
            AND ((i_dt_due IS NOT NULL) AND i_dt_due != '')
            AND ((i_dt_due IS NOT NULL) AND i_dt_due != '' AND (TO_DATE(i_dt_due, "%Y-%m-%d") > CURRENT))
        THEN

            IF
                (
                    -- NOT EXISTS (SELECT 1 FROM rms_rilt WHERE lit_no = i_lit_no)
                    -- AND
                    NOT EXISTS (SELECT 1 FROM rms_rilt WHERE lit_item_ref = i_lit_item_ref)
                )
            THEN
                LET o_result = 1;
            ELSE
                LET o_result = -2; -- DuplicateDataExternal
            END IF;

        ELSE
            LET o_result = -1; -- InvalidFormatExternal
        END IF;

        IF 
            o_result > 0
        THEN
            INSERT INTO rms_rilt
                (lit_no, lit_item_ref, lit_amt, entity_type, entity_no, dt_due, dt_created, dt_modified, created_by, modified_by, status)
            VALUES
                (i_lit_no, i_lit_item_ref, i_lit_amount, i_entity_type, i_entity_no, TO_DATE(i_dt_due, "%Y-%m-%d"), current, current, 'SYSTEM', 'SYSTEM', 'CA');
            
            LET o_result = DBINFO('sqlca.sqlerrd1');
        END IF;

        IF
            o_result > 0
        THEN
            INSERT INTO rms_rilt_a
                (rilt_id, lit_amt_bf, lit_amt_af, dt_txn, dt_created, dt_modified, created_by, modified_by, status)
            VALUES
                (o_result, 0, i_lit_amount, current, current, current, 'SYSTEM', 'SYSTEM', 'CA');

            LET o_result = DBINFO('sqlca.sqlerrd1');
        END IF;

    IF o_result > 0 THEN
        LET o_result = 1; -- SuccessResponseExternal
        COMMIT WORK;
    ELSE
        ROLLBACK WORK;
    END IF;

    RETURN o_result;

END PROCEDURE;

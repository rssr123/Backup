drop procedure sp_insfmsmtth;

create or replace procedure sp_insfmsmtth(i_customer nvarchar(100), i_inv_dt date)

returning int8 as hid;

define o_batch_no       nvarchar(30);
define o_hid            int8;
define o_result         int;
define runno            nvarchar(30);
define o_fms_branch     nvarchar(100);

on exception
	rollback work;
end exception;

    -- check got overnight anot
    execute procedure sp_getrunno('ARIV') into o_result;

    begin work;
    set lock mode to wait 30;

    /* UPDATE last running no */
    UPDATE rms_runno
    SET last_runno = last_runno +1
    WHERE runno_cd = 'ARIV';

    -- get running number
	SELECT last_runno
	INTO o_result
	FROM rms_runno
	WHERE runno_cd = 'ARIV';

	-- Generate unique receipt number
	LET o_batch_no = 'ARIV' || TO_CHAR(TODAY, '%Y%m%d') || LPAD(o_result, 6, '0');

    SELECT config_value into o_fms_branch from rms_config where config_cd = 'FMSBranch';

    -- insert rms_fms_ari_h
    -- geo: dt_sent = current
    INSERT INTO rms_fms_ari_h(type, link_branch, amt, cust, rms_batch_no, 
                                dt_sent, desc, attr_ext_sys, resp_attr_ext_sys, fms_ref_no, 
                                resp_co, resp_status, resp_msg, resp_dt, 
                                dt_created, dt_modified, created_by, modified_by, status, retry, inv_dt) 
        VALUES('Invoice', o_fms_branch, 0, i_customer, o_batch_no 
                , current, 'Collection with Receipt', 'RMS', null, null
                , null, null, null, null, current, current, 'system', 'system', 'A', 0, i_inv_dt);

    let o_hid = dbinfo('serial8');

    commit work;

return o_hid;

end procedure;

DROP PROCEDURE sp_getfmsmtt();

create or REPLACE PROCEDURE sp_getfmsmtt()
returning int8 as mtt_pg_id, decimal(16,2) as pg_pymt_amt, int as qty, nvarchar(100) as item_desc,
            decimal(16,2) as unit_fee, nvarchar(100) as rcpt_no, nvarchar(100) as cust_nm, nvarchar(100) as entity_nm, nvarchar(100) as entity_no,
            nvarchar(100) as entity_type, decimal(16,2) as gross_amt, --nvarchar(100) as fee_detail_id
            int8 as fee_detail_pk, nvarchar(100) as pg_pymt_method,
            decimal(16,2) as tax_amt,nvarchar(100) as customer,nvarchar(100) as item_ref_no, nvarchar(100) as cp_no,
            decimal(16,2) as disc_amt, nvarchar(10) as dps_id, nvarchar(30) as dps_task,decimal(16,2) as net_amt, datetime year to fraction(3) as inv_dt;

define o_mtt_pg_id int8;
define o_pg_pymt_amt decimal(16,2);
define o_qty int;
define o_item_desc nvarchar(250);
define o_unit_fee decimal(16,2);
define o_rcpt_no nvarchar(30);
define o_cust_nm nvarchar(250);
define o_entity_nm nvarchar(100);
define o_entity_no nvarchar(100);
define o_entity_type nvarchar(1);
define o_gross_amt decimal(16,2);
define o_fee_detail_id nvarchar(100);
define o_fee_detail_pk int8;
define o_pg_pymt_method nvarchar(50);
define o_tax_amt decimal(16,2);

define o_product nvarchar(100);
define o_coa1 nvarchar(100);
define o_customer nvarchar(100);
define o_item_ref_no nvarchar(100);
define o_cp_no nvarchar(100);

define o_disc_amt decimal(16,2);
define o_dps_id nvarchar(10);
define o_dps_task nvarchar(30);

define o_net_amt decimal(16,2);

define ari_limit nvarchar(100);
define temp_date date;

    -- get limit per payment
    select config_value
    into ari_limit
    from rms_config
    where config_cd = 'FMS_ARI_Details';

    -- filter in rms_fms_arrb's mtt record
    -- get the most earliest date
    select min(date(pg_resp_time)) as resp_time
    into temp_date
    from rms_mtt_pg
    where (in_fms_posting = 0 or in_fms_posting is null)
    and pg_txn_status=0
    and date(pg_resp_time) <= today -1;
    --and date(pg_resp_time) <= today;

    -- customer field
    select acct_cd
    into o_customer
    from rms_fms_acct
    where acct_nm ='Revenue';

foreach c_ari for
    -- filter out the limit
    SELECT t1.mtt_pg_id,
       t1.pg_pymt_amt,
       t2.qty,
       t2.item_desc,
       t2.unit_fee,
       t4.rcpt_no,
       t3.cust_nm,
       t2.entity_nm,
       t2.entity_no,
       t2.entity_type,
       t2.gross_amt,
--       t2.fee_detail_id, --change to use pk
       t2.fee_detail_pk,
       t1.pg_pymt_method,
       t2.tax_amt,
       t2.item_ref_no,
       t2.cp_no,
       t2.disc_amt,
       t2.dps_id,
       t2.dps_task,
       t2.net_amt
    into o_mtt_pg_id,o_pg_pymt_amt,o_qty,o_item_desc,o_unit_fee,o_rcpt_no,o_cust_nm,o_entity_nm,o_entity_no,o_entity_type
        ,o_gross_amt,o_fee_detail_pk,o_pg_pymt_method,o_tax_amt,o_item_ref_no,o_cp_no,o_disc_amt,o_dps_id,o_dps_task,o_net_amt
    FROM (
        SELECT FIRST ari_limit
            mtt_pg_id,
            mtt_id,
            pg_pymt_amt,
            pg_pymt_method,
            pg_resp_time
        FROM rms_mtt_pg
        WHERE DATE(pg_resp_time) = temp_date
          AND (in_fms_posting = 0 OR in_fms_posting IS NULL)
          AND pg_txn_status = 0
        ORDER BY pg_resp_time
    ) t1
    INNER JOIN rms_mtt_item t2 ON t1.mtt_id = t2.mtt_id
    INNER JOIN rms_mtt t3 ON t1.mtt_id = t3.mtt_id
    LEFT JOIN rms_mtt_rcpt t4 ON t1.mtt_pg_id = t4.mtt_pg_id
    WHERE t4.rcpt_status = 'A'
    ORDER BY t1.pg_resp_time

    return o_mtt_pg_id,o_pg_pymt_amt,o_qty,o_item_desc,o_unit_fee,o_rcpt_no,o_cust_nm,o_entity_nm,o_entity_no,o_entity_type
    ,o_gross_amt,o_fee_detail_pk,o_pg_pymt_method,o_tax_amt,o_customer,o_item_ref_no,o_cp_no,o_disc_amt,o_dps_id,o_dps_task,o_net_amt,temp_date with resume;

end foreach;

end procedure;

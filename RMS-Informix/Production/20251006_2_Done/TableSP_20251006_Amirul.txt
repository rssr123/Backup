drop procedure sp_getserviceproviderpayment

go


CREATE OR REPLACE PROCEDURE sp_getserviceproviderpayment(
    i_page INTEGER, 
    i_size INTEGER, 
    i_profile_nm NVARCHAR(100), 
    i_ag_bil_no NVARCHAR(100), 
    i_cust_email NVARCHAR(100), 
    i_total_amt_payable DECIMAL(16, 2), 
    i_date_collection_fr DATETIME YEAR TO SECOND, 
    i_date_collection_to DATETIME YEAR TO SECOND, 
    i_pymt_status NVARCHAR(50),
    i_dt_pymt_fr DATETIME YEAR TO SECOND, 
    i_dt_pymt_to DATETIME YEAR TO SECOND, 
    i_date_email_sent_fr DATETIME YEAR TO SECOND, 
    i_date_email_sent_to DATETIME YEAR TO SECOND
)
RETURNING 

	INTEGER as ag_bil,
    NVARCHAR(100) AS profile_nm,
    NVARCHAR(100) AS ag_bil_no,
    NVARCHAR(100) AS cust_email,
    DECIMAL(16, 2) AS total_amt_payable,
    DATETIME YEAR TO SECOND AS date_collection,
    NVARCHAR(50) AS pymt_status,
    DATETIME YEAR TO SECOND AS dt_pymt,
    DATETIME YEAR TO SECOND AS date_email_sent,
    NVARCHAR(50) AS order_status,
    INTEGER AS total;

    -- Define variables
   	define o_ag_bil INTEGER;
    DEFINE o_profile_nm NVARCHAR(100);
   	DEFINE o_ag_bil_no NVARCHAR(100);
    DEFINE o_cust_email NVARCHAR(100);
    DEFINE o_total_amt_payable DECIMAL(16, 2);
    DEFINE o_date_collection DATETIME YEAR TO SECOND;
   	DEFINE o_pymt_status NVARCHAR(50);
    DEFINE o_dt_pymt DATETIME YEAR TO SECOND;
    DEFINE o_date_email_sent DATETIME YEAR TO SECOND;
    DEFINE calculated_page INTEGER;
   	DEFINE o_order_status NVARCHAR(50);
    DEFINE total_records INTEGER;

    -- Validate inputs
    IF (i_page IS NULL OR i_page < 1) OR (i_size IS NULL OR i_size < 1) THEN 
        RAISE EXCEPTION -746, 0, "i_page or i_size is null or invalid integer";
    END IF;

    -- Calculate offset for pagination
    LET calculated_page = (i_page - 1) * i_size;

    -- Count total matching records
    LET total_records = (
        SELECT COUNT(1) 
        FROM rms_ag_bil t1
        left join rms_param t2 on t1.pymt_status = t2.param_cd
		left join rms_ag_pf t3 on t1.ag_pf_id = t3.ag_pf_id
		left join rms_mtt t4 on t1.ag_bil_no = t4.orn_no
        WHERE (i_profile_nm IS NULL OR t3.profile_nm LIKE '%' || i_profile_nm || '%')
          AND (i_ag_bil_no IS NULL OR t1.ag_bil_no LIKE '%' || i_ag_bil_no || '%')
          AND (i_cust_email IS NULL OR t1.cust_email LIKE '%' || i_cust_email || '%')
          AND (i_total_amt_payable IS NULL OR t1.total_amt_payable = i_total_amt_payable)
          AND (i_date_collection_fr IS NULL OR t1.date_collection >= i_date_collection_fr)
          AND (i_date_collection_to IS NULL OR t1.date_collection <= i_date_collection_to)
          AND (i_pymt_status IS NULL OR t1.pymt_status LIKE '%' || i_pymt_status || '%')
          AND (i_dt_pymt_fr IS NULL OR t1.dt_pymt >= i_dt_pymt_fr)
          AND (i_dt_pymt_to IS NULL OR t1.dt_pymt <= i_dt_pymt_to)
          AND (i_date_email_sent_fr IS NULL OR t1.date_email_sent >= i_date_email_sent_fr)
          AND (i_date_email_sent_to IS NULL OR t1.date_email_sent <= i_date_email_sent_to)
          and  t2.param_grp_nm = 'cc-pymt-status'
    );

    -- Fetch paginated results
    FOREACH c_serviceproviderpayment FOR
        SELECT SKIP calculated_page LIMIT i_size
        	t1.ag_bil,
            t3.profile_nm,
            t1.ag_bil_no,
            t1.cust_email,
            t1.total_amt_payable,
            t1.date_collection,
            t2.nm_en,
            t1.dt_pymt,
            t1.date_email_sent,
            t4.order_status
        INTO 
        	o_ag_bil,
            o_profile_nm,
            o_ag_bil_no,
            o_cust_email,
            o_total_amt_payable,
            o_date_collection,
            o_pymt_status,
            o_dt_pymt,
            o_date_email_sent,
            o_order_status
        FROM rms_ag_bil t1
        left join rms_param t2 on t1.pymt_status = t2.param_cd
        left join rms_ag_pf t3 on t1.ag_pf_id = t3.ag_pf_id
        left join rms_mtt t4 on t1.ag_bil_no = t4.orn_no
        WHERE (i_profile_nm IS NULL OR t3.profile_nm LIKE '%' || i_profile_nm || '%')
          AND (i_ag_bil_no IS NULL OR t1.ag_bil_no LIKE '%' || i_ag_bil_no || '%')
          AND (i_cust_email IS NULL OR t1.cust_email LIKE '%' || i_cust_email || '%')
          AND (i_total_amt_payable IS NULL OR t1.total_amt_payable = i_total_amt_payable)
          AND (i_date_collection_fr IS NULL OR t1.date_collection >= i_date_collection_fr)
          AND (i_date_collection_to IS NULL OR t1.date_collection <= i_date_collection_to)
          AND (i_pymt_status IS NULL OR t1.pymt_status LIKE '%' || i_pymt_status || '%')
          AND (i_dt_pymt_fr IS NULL OR t1.dt_pymt >= i_dt_pymt_fr)
          AND (i_dt_pymt_to IS NULL OR t1.dt_pymt <= i_dt_pymt_to)
          AND (i_date_email_sent_fr IS NULL OR t1.date_email_sent >= i_date_email_sent_fr)
          AND (i_date_email_sent_to IS NULL OR t1.date_email_sent <= i_date_email_sent_to)
          and t2.param_grp_nm = 'cc-pymt-status'
        ORDER BY t1.dt_created desc
        RETURN 
        	o_ag_bil,
            o_profile_nm,
            o_ag_bil_no,
            o_cust_email,
            o_total_amt_payable,
            o_date_collection,
            o_pymt_status,
            o_dt_pymt,
            o_date_email_sent,
            o_order_status,
            total_records WITH RESUME;
    END FOREACH;
END PROCEDURE;

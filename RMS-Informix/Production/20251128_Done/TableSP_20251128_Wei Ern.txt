CREATE OR REPLACE PROCEDURE sp_insfmsrcpgmtt_f(	i_crmemo_hid INT8,
											    i_mtt_pg_id INT8, i_is_drmemo integer)
RETURNING INT AS result;

    DEFINE o_result INT;
    DEFINE o_exist INT;
    DEFINE o_fms_ref_no1 NVARCHAR(50);
    DEFINE v_doc_ty NVARCHAR(20);

    on exception
		rollback work;
	end exception;
	 
	begin work;
	set lock mode to wait 30;

    LET v_doc_ty = CASE 
        WHEN i_is_drmemo = 1 THEN 'Debit Memo'
        ELSE 'Invoice'
    END;

    -- create temp table
    CREATE TEMP TABLE tempA(
        o_fms_ref_no NVARCHAR(50)
    );

    -- populate temp table
    if i_is_drmemo = 0 then
        INSERT INTO tempA (o_fms_ref_no)
        SELECT DISTINCT t3.fms_ref_no
        FROM rms_fms_ari_h t3
        INNER JOIN rms_fms_ari_b t2 ON t3.ari_hid = t2.ari_hid
        WHERE t2.mtt_pg_id = i_mtt_pg_id AND t3.status = 'A';
    end if;

    if i_is_drmemo = 1 then
        INSERT INTO tempA (o_fms_ref_no)
        SELECT first 1 t1.fms_ref_no
        FROM rms_fms_drmemo_h t1
        INNER JOIN rms_fms_drmemo_b t2 ON t1.drmemo_hid = t2.drmemo_hid
        INNER JOIN rms_fms_drmemo_f t3 ON t1.drmemo_hid = t3.drmemo_hid
        WHERE t2.mtt_pg_id = i_mtt_pg_id AND t3.status = 'A' order by t1.dt_created;
    end if;
--826c9487-b78c-47bc-b02a-2
   
   	LET o_result = 0;
   
    -- loop tempA and check if fms_ref_no exists in the footer
    FOREACH 
    	SELECT o_fms_ref_no 
    	INTO o_fms_ref_no1
    	FROM tempA
    	
        -- check if the record exists
        SELECT COUNT(1) 
        INTO o_exist
        FROM rms_fms_crmemo_f
        WHERE crmemo_hid = i_crmemo_hid
        AND fms_ref_no = o_fms_ref_no1;

        -- Insert if the record does not exist
        IF o_exist = 0 THEN
            INSERT INTO rms_fms_crmemo_f (
                crmemo_hid, fms_ref_no, doc_ty, dt_created, dt_modified, created_by,
                modified_by, status
            ) VALUES (
                i_crmemo_hid, o_fms_ref_no1, v_doc_ty, CURRENT, CURRENT, 'system',
                'system', 'A'
            );
           	LET o_result = o_result + 1;
        END IF;
    END FOREACH;

--    LET o_result = DBINFO('sqlca.sqlerrd2');

    -- drop temp table
    DROP TABLE tempA;

    IF (o_result > 0) THEN
        COMMIT WORK;
        RETURN o_result;
    ELSE
        ROLLBACK WORK;
        RETURN -1;
    END IF;

END PROCEDURE;

CREATE OR REPLACE PROCEDURE sp_updpgmtttxn()
returning int8 as result;
 
define o_result int;
define o_exist int;
define o_compare_range int;
define o_dt_settlement date;

on exception
	rollback work;
end exception;
 
-- Set a default value
LET o_result = '0';
 
begin work;
set lock mode to wait 30;

-- create a temporary table
create temp table temppgmtt (
    rc_pgmtt_id int8,
    rc_pg_id int8,
    sub_criteria nvarchar(10)
);
 
-- check exist
select count(1)
into o_exist
from rms_rc_pgmtt
where sub_criteria = 'NF';

if o_exist > 0 then

    -- select statement
    insert into temppgmtt (rc_pgmtt_id,rc_pg_id,sub_criteria)
    SELECT 
        rms_rc_pgmtt.rc_pgmtt_id, 
        rms_rc_pgmtt.rc_pg_id,
        CASE 
            WHEN rms_rc_pgtxn.txn_id IS NULL THEN 'NFP'
            ELSE 'FIP' --Found in Prev. not need care task status.
--            WHEN rms_rc_pg.task_status = 'CS' THEN 'CIP'
--            ELSE 'NCP'
        END as Status
    FROM 
        rms_rc_pgmtt
    Inner JOIN 
        rms_mtt_pg ON rms_rc_pgmtt.mtt_pg_id = rms_mtt_pg.mtt_pg_id  and (rms_mtt_pg.pg_txn_status=0 or rms_mtt_pg.pg_txn_status=2)
    LEFT JOIN 
        rms_rc_pgtxn ON rms_mtt_pg.pg_txn_id = rms_rc_pgtxn.txn_id and rms_rc_pgtxn.status = 'A'
    LEFT JOIN 
        rms_rc_pg ON rms_rc_pgtxn.rc_pg_id = rms_rc_pg.rc_pg_id
    WHERE 
        rms_rc_pgmtt.sub_criteria = 'NF';

    -- update
    UPDATE 
        rms_rc_pgmtt
    SET 
        rms_rc_pgmtt.sub_criteria = temppgmtt.sub_criteria
    FROM 
        temppgmtt
    WHERE 
        rms_rc_pgmtt.rc_pgmtt_id = temppgmtt.rc_pgmtt_id;
 
    -- get update count
    let o_result = dbinfo('sqlca.sqlerrd2');

    -- update rms_rc_pg' recon status
    update rms_rc_pg
    set recon_status = 'PSSCR'
    where rc_pg_id in (select distinct rc_pg_id from temppgmtt);

drop table temppgmtt;
 
end if;
 
commit work;
 
if o_result > 0 then
    return o_result;
else
    return -1;
end if;
 
end procedure
;

CREATE OR REPLACE PROCEDURE sp_getfmsrcbank()
returning 
    int8 as rc_bank_id,
    int8 as rc_pg_id,
    decimal(16,2) as credit,
    decimal(16,2) as mdr_amt,
    nvarchar(10) as acct_cd,
    nvarchar(10) as cust;

define o_rc_bank_id int8;
define o_rc_pg_id int8;
define o_credit decimal(16,2);
define o_mdr_amt decimal(16,2);
define o_acct_cd nvarchar(10);
define o_cust nvarchar(10);

let o_acct_cd = (select first 1 acct_cd from rms_fms_acct where acct_nm = 'AR Receipt''s Cash Account');
let o_cust = (select first 1 acct_cd from rms_fms_acct where acct_nm = 'Revenue');

foreach c_arr for
    -- MDR aggregation (PG side)
    with mdr_totals as (
        select 
            b.rc_bank_id,
            b.rc_pg_id,
            sum(p.mdr_amt) as total_mdr_amt
        from rms_rc_bank b
        join rms_rc_pgtxn p on p.rc_pg_id = b.rc_pg_id
        where (b.in_fms_posting = 0 or b.in_fms_posting is null)
        and date(b.dt_created) <= today
        and b.recon_status = 'SBSR'
        group by b.rc_bank_id, b.rc_pg_id
    ),
    -- CREDIT aggregation (BankTxn side, based on stmt_no pattern match)
    credit_totals as (
        select 
            t2.rc_bank_id,
            t2.rc_pg_id,
            sum(t1.credit) as total_credit
        from rms_rc_banktxn t1
        join rms_rc_bank t2 
            on t2.rc_bank_id = t1.rc_bank_id
            and t2.recon_status = 'SBSR'
        join rms_rc_pg t3 
            on t3.rc_pg_id = t2.rc_pg_id 
            and t1.txn_ref like t3.stmt_no || '%'
        group by t2.rc_bank_id, t2.rc_pg_id
    )

    -- Final combined result
    select 
        m.rc_bank_id,
        m.rc_pg_id,
        m.total_mdr_amt,
        c.total_credit
        into o_rc_bank_id, o_rc_pg_id, o_mdr_amt, o_credit
    from mdr_totals m
    left join credit_totals c 
        on c.rc_bank_id = m.rc_bank_id 
        and c.rc_pg_id = m.rc_pg_id
    order by m.rc_bank_id

    return o_rc_bank_id, o_rc_pg_id, o_credit, o_mdr_amt, o_acct_cd, o_cust with resume;
end foreach;

end procedure;

                    i_tax_amt, current, current, 'system', 'system', 
                    'A', i_fee_detail_id, i_mtt_pg_id, next_line_nbr, i_disc_amt, i_deposit_id, i_deposit_task);
            end if;
        -- Check for RILT
        elif i_lit_item_ref is not null and i_lit_item_ref != '' then 
            select acct_cd
            into coa1_lt
            from rms_fms_acct
            where acct_nm = 'RI/LT';

             select coalesce(max(line_nbr), 0) + 1 into next_line_nbr
             from rms_fms_ari_b
             where ari_hid = i_hid;

            INSERT INTO rms_fms_ari_b(ari_hid, acct, branch, qty, sub_acct,

                    i_tax_amt, current, current, 'system', 'system', 
                    'A', i_fee_detail_id, i_mtt_pg_id, next_line_nbr, i_disc_amt, i_deposit_id, i_deposit_task, i_bil_child_id);

end if;

    let o_result = dbinfo('serial8');

    if i_flag = 1 then
    
        update rms_mtt_pg
        set in_fms_posting = 1,
        dt_modified = current
        where mtt_pg_id=i_mtt_pg_id;

        -- update rms_fms_ari_h

    -- Only for CP with Disc Amt
    /*if o_accr_amt > i_net_amt then

end procedure;

DROP PROCEDURE sp_getfmsmtt;
create or REPLACE PROCEDURE sp_getfmsmtt()
returning int8 as mtt_pg_id, decimal(16,2) as pg_pymt_amt, int as qty, nvarchar(100) as item_desc,
            decimal(16,2) as unit_fee, nvarchar(100) as rcpt_no, nvarchar(100) as cust_nm, nvarchar(100) as entity_nm, nvarchar(100) as entity_no,
            nvarchar(100) as entity_type, decimal(16,2) as gross_amt, --nvarchar(100) as fee_detail_id
            int8 as fee_detail_pk, nvarchar(100) as pg_pymt_method,
            decimal(16,2) as tax_amt,nvarchar(100) as customer,nvarchar(100) as item_ref_no, nvarchar(100) as cp_no,
            decimal(16,2) as disc_amt, nvarchar(10) as dps_id, nvarchar(30) as dps_task,decimal(16,2) as net_amt, datetime year to fraction(3) as inv_dt, NVARCHAR(100) as lit_item_ref;

define o_mtt_pg_id int8;
define o_pg_pymt_amt decimal(16,2);
define o_qty int;
define o_item_desc nvarchar(250);
define o_unit_fee decimal(16,2);
define o_rcpt_no nvarchar(30);
define o_cust_nm nvarchar(250);
define o_entity_nm nvarchar(100);
define o_entity_no nvarchar(100);
define o_entity_type nvarchar(1);
define o_gross_amt decimal(16,2);
define o_fee_detail_id nvarchar(100);
define o_fee_detail_pk int8;
define o_pg_pymt_method nvarchar(50);
define o_tax_amt decimal(16,2);

define o_product nvarchar(100);
define o_coa1 nvarchar(100);
define o_customer nvarchar(100);
define o_item_ref_no nvarchar(100);
define o_cp_no nvarchar(100);

define o_disc_amt decimal(16,2);
define o_dps_id nvarchar(10);
define o_dps_task nvarchar(30);

define o_net_amt decimal(16,2);

define ari_limit nvarchar(100);
define temp_date date;
define o_lit_item_ref nvarchar(100);

    -- get limit per payment
    select config_value
    into ari_limit
    from rms_config
    where config_cd = 'FMS_ARI_Details';

    -- filter in rms_fms_arrb's mtt record
    -- get the most earliest date
    select min(date(pg_resp_time)) as resp_time
    into temp_date
    from rms_mtt_pg
    where (in_fms_posting = 0 or in_fms_posting is null)
    and (pg_txn_status=0 or pg_txn_status=2)
    and date(pg_resp_time) <= today -1;
    --and date(pg_resp_time) <= today;

    -- customer field
    select acct_cd
    into o_customer
    from rms_fms_acct
    where acct_nm ='Revenue';

foreach c_ari for
    -- filter out the limit
    SELECT t1.mtt_pg_id,
       t1.pg_pymt_amt,
       t2.qty,
       t2.item_desc,
       t2.unit_fee,
       NVL(t4.rcpt_no,''),
       t3.cust_nm,
       t2.entity_nm,
       t2.entity_no,
       t2.entity_type,
       t2.gross_amt,
--       t2.fee_detail_id, --change to use pk
       t2.fee_detail_pk,
       t1.pg_pymt_method,
       t2.tax_amt,
       t2.item_ref_no,
       t2.cp_no,
       t2.disc_amt,
       t2.dps_id,
       t2.dps_task,
       t2.net_amt,
       t2.lit_item_ref
    into o_mtt_pg_id,o_pg_pymt_amt,o_qty,o_item_desc,o_unit_fee,o_rcpt_no,o_cust_nm,o_entity_nm,o_entity_no,o_entity_type
        ,o_gross_amt,o_fee_detail_pk,o_pg_pymt_method,o_tax_amt,o_item_ref_no,o_cp_no,o_disc_amt,o_dps_id,o_dps_task,o_net_amt, o_lit_item_ref
    FROM (
        SELECT FIRST ari_limit
            mtt_pg_id,
            mtt_id,
            pg_pymt_amt,
            pg_pymt_method,
            pg_resp_time
        FROM rms_mtt_pg
        WHERE DATE(pg_resp_time) = temp_date
          AND (in_fms_posting = 0 OR in_fms_posting IS NULL)
          AND (pg_txn_status = 0 OR pg_txn_status = 2)
        ORDER BY pg_resp_time
    ) t1
    INNER JOIN rms_mtt_item t2 ON t1.mtt_id = t2.mtt_id
    INNER JOIN rms_mtt t3 ON t1.mtt_id = t3.mtt_id
    LEFT JOIN rms_mtt_rcpt t4 ON t1.mtt_pg_id = t4.mtt_pg_id AND t4.rcpt_status = 'A'
    -- WHERE t4.rcpt_status = 'A'
    ORDER BY t1.pg_resp_time

    return o_mtt_pg_id,o_pg_pymt_amt,o_qty,o_item_desc,o_unit_fee,o_rcpt_no,o_cust_nm,o_entity_nm,o_entity_no,o_entity_type
    ,o_gross_amt,o_fee_detail_pk,o_pg_pymt_method,o_tax_amt,o_customer,o_item_ref_no,o_cp_no,o_disc_amt,o_dps_id,o_dps_task,o_net_amt,temp_date, o_lit_item_ref with resume;

end foreach;

end procedure;

    nvarchar(10) as acct_cd,
    nvarchar(10) as cust,
    nvarchar(100) as fms_ref_no;

define o_mdr_amt decimal(16,2);
define o_fms_ref_no nvarchar(100);
define o_acct_cd nvarchar(10);
define o_cust nvarchar(10);

-- let o_cust = (select first 1 acct_cd from rms_fms_acct where acct_nm = 'Payment Gateway');

if i_rc_pg_id = 0 then
    UPDATE rms_rc_pgtxn t1
    SET mtt_pg_id = (
        SELECT t2.mtt_pg_id
        FROM rms_mtt_pg t2
        WHERE t1.txn_id LIKE '%' || t2.pg_pymt_id || '%'
          AND t2.pg_pymt_id IS NOT NULL
    )
    WHERE mtt_pg_id IS NULL
      AND 1 = (
        SELECT COUNT(*)
        FROM rms_mtt_pg t2
        WHERE t1.txn_id LIKE '%' || t2.pg_pymt_id || '%'
          AND t2.pg_pymt_id IS NOT NULL
      );

    foreach c_arr for

      -- select t1.amt, t1.fms_ref_no, nvl(sum(t4.mdr_amt),0)
      select t1.amt, t1.fms_ref_no, 0, t1.cust
      into o_credit, o_fms_ref_no, o_mdr_amt, o_cust
      from rms_fms_drmemo_h t1 
      left join rms_fms_drmemo_b t2 on t1.drmemo_hid = t2.drmemo_hid 
      left join rms_fms_drmemo_f t3 on t1.drmemo_hid = t3.drmemo_hid
      left join rms_rc_pgtxn t4 on t2.rc_pgtxn_id = t4.rc_pgtxn_id
      -- left join rms_rc_pgtxn t4 on t2.mtt_pg_id = t4.mtt_pg_id
      where (t2.in_fms_posting = 0 or t2.in_fms_posting is null)
        and t1.status = 'A' 
        --and t4.rc_pg_id = i_rc_pg_id
        -- and t4.mtt_pg_id is not null
      group by t1.amt, t1.fms_ref_no, t4.mdr_amt, t1.cust

        return o_credit, o_mdr_amt, o_acct_cd, o_cust, o_fms_ref_no with resume;
    end foreach;
else
    UPDATE rms_rc_pgtxn t1
    SET mtt_pg_id = (
        SELECT t2.mtt_pg_id
        FROM rms_mtt_pg t2
        WHERE t1.txn_id LIKE '%' || t2.pg_pymt_id || '%'
          AND t2.pg_pymt_id IS NOT NULL
    )
    WHERE mtt_pg_id IS NULL
      AND 1 = (
        SELECT COUNT(*)
        FROM rms_mtt_pg t2
        WHERE t1.txn_id LIKE '%' || t2.pg_pymt_id || '%'
          AND t2.pg_pymt_id IS NOT NULL
      );

    foreach c_arr for

      -- select t1.amt, t1.fms_ref_no, nvl(sum(t4.mdr_amt),0)
      select t1.amt, t1.fms_ref_no, 0, t1.cust
      into o_credit, o_fms_ref_no, o_mdr_amt, o_cust
      from rms_fms_drmemo_h t1 
      left join rms_fms_drmemo_b t2 on t1.drmemo_hid = t2.drmemo_hid

      left join rms_rc_pgtxn t4 on t2.rc_pgtxn_id = t4.rc_pgtxn_id
      -- left join rms_rc_pgtxn t4 on t2.mtt_pg_id = t4.mtt_pg_id
      where (t2.in_fms_posting = 0 or t2.in_fms_posting is null)
        and t1.status = 'A' 
        and t4.rc_pg_id = i_rc_pg_id
        -- and t4.mtt_pg_id is not null
      group by t1.amt, t1.fms_ref_no, t4.mdr_amt, t1.cust

        return o_credit, o_mdr_amt, o_acct_cd, o_cust, o_fms_ref_no with resume;
    end foreach;
end if;



end procedure;

create or replace procedure sp_updcccasetaskstatus(i_task_no nvarchar(50), i_task_status nvarchar(10), i_assign_to nvarchar(255),
													i_user nvarchar(25), i_remark lvarchar(4000), i_msg_type nvarchar(10))
	returning integer as result;

	define i_cc_case_id int;
	define o_result int;
	define i_running_time datetime year to fraction(5);
	--define i_assign_to nvarchar(25);
	define i_pick_up nvarchar(25);
	define i_dt_pickup datetime year to fraction(5);

	on exception
		rollback work;
	end exception;

	begin work;
	set lock mode to wait 30;

	let i_cc_case_id = 0;
	let o_result = 0;
	let i_running_time = current;
	
	select limit 1 cc_case_id, pick_up into i_cc_case_id, i_pick_up from rms_cc_case where task_no = i_task_no and status = 'A';
	if i_assign_to is null or i_assign_to == '' then
		select limit 1 role_name into i_assign_to from rms_param where param_cd = i_task_status and param_grp_nm = 'cc-case';
	end if;

	if i_cc_case_id == 0 or i_cc_case_id is null then 
		let o_result = -103;
	else
		if i_pick_up is null or i_user != i_pick_up then
			update rms_cc_case set pick_up = i_user,
								dt_pickup = i_running_time,
								dt_modified = i_running_time,
								modified_by = i_user
			where cc_case_id = i_cc_case_id;
		
			let o_result = dbinfo('sqlca.sqlerrd2');
			if o_result == 0 or o_result is null then
				let o_result = -1;
			else
				let o_result = 1;
			end if;

			if o_result == 1 then 
				execute procedure sp_inscccasehist(i_cc_case_id, i_running_time, i_user, 
					(case when i_pick_up is null then 'Direct Task pick up' else 'Task override' end), 'PU') into o_result;
				if o_result < 1 then
					rollback work;
				elif o_result is null then
					rollback work;
					let o_result = -2;
				end if;
			end if;
		end if;
		if o_result > -1 then
			create temp table tmpHistory(
				ts nvarchar(100),
				pu nvarchar(25),
				mt nvarchar(10),
				dc datetime year to fraction(5)
			) with no log;
		
			insert into tmpHistory
				select a.task_status, a.pick_up, m.msg_type, a.dt_created
					from rms_cc_case_a a left join rms_cc_msg m on a.cc_case_a_id = m.cc_case_a_id
				where a.cc_case_id = i_cc_case_id and a.modified_by != 'system' order by a.dt_created desc;
		
			let i_pick_up = null;
			let i_dt_pickup = null;
			
			if i_task_status = 'IP' or i_task_status = 'RQ' or i_task_status = 'CMI' or i_task_status = 'DMI' or i_task_status = 'I' then 
				let o_result = (select count(1) from tmpHistory where pu is not null and pu != 'system' and (ts = 'IP' or ts = 'PD'));
				if o_result > 0 then
					select limit 1 pu into i_pick_up from tmpHistory where pu is not null and pu != 'system' and (ts = 'IP' or ts = 'PD');
					let i_dt_pickup = i_running_time;
				end if;
			elif i_task_status = 'PL' then 
				let o_result = (select count(1) from tmpHistory where pu is not null and pu != 'system' and ts = 'PL');
				if o_result > 0 then
					select limit 1 pu into i_pick_up from tmpHistory where pu is not null and pu != 'system' and ts = 'PL';
					let i_dt_pickup = i_running_time;
				end if;
			elif i_task_status = 'PCO' then 
				let o_result = (select count(1) from tmpHistory where pu is not null and pu != 'system' and ts = 'PCO');
				if o_result > 0 then
					select limit 1 pu into i_pick_up from tmpHistory where pu is not null and pu != 'system' and ts = 'PCO';
					let i_dt_pickup = i_running_time;
				end if;
			elif i_task_status = 'C' then 
				let i_pick_up = null;
			end if;
					
			drop table tmpHistory;
			
			let i_running_time = current;
					
			update rms_cc_case set task_status = i_task_status,
									assign_to = i_assign_to,
									pick_up  = i_pick_up,
									dt_pickup = i_dt_pickup,
									dt_modified = i_running_time,
									modified_by = i_user
			where cc_case_id = i_cc_case_id;
		
			let o_result = dbinfo('sqlca.sqlerrd2');
			if o_result == 0 or o_result is null then
				let o_result = -3;
			else
				let o_result = 1;
			end if;
	
			if o_result == 1 then 
				execute procedure sp_inscccasehist(i_cc_case_id, i_running_time, i_user, i_remark, i_msg_type) into o_result;
				if o_result < 1 then
					rollback work;
				elif o_result is null then
					rollback work;
					let o_result = -4;
				end if;
			end if;
		end if;
	end if;
	commit work;
	return o_result;
end procedure;

	define billing_mthd_to_use nvarchar(255);
	define i_running_time datetime year to fraction(5);
	define i_assign_to_t nvarchar(255);

	on exception
		rollback work;
	end exception;
	begin work;
	set lock mode to wait 30;
	
	let i_running_time = current;

	if i_origin_table = 'rms_rtt_wf' then
	
		-- handle refund scenario (update rms_rtt_wf, insert a new record into rms_rtt_wf_hist)
		-- update rms_rtt_wf
		update rms_rtt_wf t1
		set t1.pickup_by = i_pickup_person,
	        t1.modified_by = i_pickup_person,
			t1.date_pick = i_running_time,
	        t1.dt_modified = i_running_time
		where (i_pk = t1.rtt_wf_id);
		
		-- insert a new record into the hist table
		-- create a temporary table with the necessary structure
		create temp table temp_rtt_wf_hist (
		    rtt_wf_id int8,
		    rtt_app_no nvarchar(255),
		    orn_no nvarchar(255),
		    txn_id nvarchar(255),
		    action nvarchar(50),
		    dt_action datetime year to second,
		    requested_by nvarchar(255),
		    dt_created datetime year to second,
		    dt_modified datetime year to second,
		    created_by nvarchar(255),
		    modified_by nvarchar(25),
		    rtt_status nvarchar(10),
	        assign_to nvarchar(255)
		) with no log;
		
		-- populate the temporary table with ordered data
		insert into temp_rtt_wf_hist
		select rtt_wf_id, rtt_app_no, orn_no, txn_id, action, dt_action, requested_by,
		       dt_created, dt_modified, created_by, modified_by, rtt_status, assign_to
		from rms_rtt_wf_hist
		where rtt_wf_id = i_pk
		order by rtt_wf_hist_id desc; -- this is crucial and to make sure the latest data is the first data matched the identifier
		
		-- fetch the first record from the temporary table
		select limit 1 rtt_wf_id, rtt_app_no, orn_no, txn_id, action, dt_action, requested_by, 
		              dt_created, dt_modified, created_by, modified_by, rtt_status, assign_to
		into rtt_wf_id_to_use, rtt_app_no_to_use, orn_no_to_use, txn_id_to_use, action_to_use,
		     dt_action_to_use, requested_by_to_use, dt_created_to_use, dt_modified_to_use,
		     created_by_to_use, modified_by_to_use, rtt_status_to_use, assign_to_to_use
		from temp_rtt_wf_hist;
		
		-- drop the temporary table to clean up
		drop table temp_rtt_wf_hist;
		
		-- insert the prepared values into the hist table
		insert into rms_rtt_wf_hist(rtt_wf_id, rtt_app_no, orn_no, txn_id, action, dt_action, requested_by, dt_created, dt_modified, created_by, modified_by, rtt_status, pickup_by, date_pick, assign_to)
		values(rtt_wf_id_to_use, rtt_app_no_to_use, orn_no_to_use, txn_id_to_use, 'Job Pick Up', dt_action_to_use, requested_by_to_use, dt_created_to_use, dt_modified_to_use, i_pickup_person, i_pickup_person, rtt_status_to_use, i_pickup_person, i_running_time, assign_to_to_use);
				
		-- return
		let o_result = 1;
	
	-- handle billing scenario
	
	-- update rms_bil or rms_bil_wf accordingly based on the guard condition, same goes to the insert new record to hist table
	-- for cancel and adjustment billing tasks
	elif i_origin_table = 'rms_bil' then
	
	    if i_pk == 0 or i_pk is null then
	        return -101; -- error code to show the bil_id doesn't exist
	    elif i_pk != 0 and i_pk is not null then
	        -- update the rms_bil
	        update rms_bil
	        set pickup_by = i_pickup_person,
	            modified_by = i_pickup_person,
	            dt_pick = i_running_time,
	            dt_modified = i_running_time
	        where bil_id = i_pk;
	       
	       	EXECUTE PROCEDURE sp_insbilhist(i_pk, i_running_time, i_pickup_person, 'Task pick up', 'PU') into o_result;
			if o_result < 1 then
				rollback work;
			elif o_result is null then
				rollback work;
	        	let o_result = -2;
			end if;
	    end if;
	
	-- for approval billing tasks
	elif i_origin_table = 'rms_bil_wf' then
	
	    if i_pk == 0 or i_pk is null then
	        return -102; -- error code to show the bil_wf_id doesn't exist
	    elif i_pk != 0 and i_pk is not null then
	        update rms_bil_wf
	        set pickup_by = i_pickup_person,
	            modified_by = i_pickup_person,
	            dt_pick = current,
	            dt_modified = current
	        where bil_wf_id = i_pk;
	       
	       	EXECUTE PROCEDURE sp_insbilhistwf(i_pk, i_running_time, i_pickup_person, 'Task pick up', 'PU') into o_result;
			if o_result < 1 then
				rollback work;
			elif o_result is null then
				rollback work;
	        	let o_result = -3;
			end if;
	    end if;
	
	elif i_origin_table = 'rms_otc_rc' then
	
		-- handle billing scenario
		-- update rms_otc_rc
		update rms_otc_rc
		set assigned_to = i_pickup_person,
	        modified_by = i_pickup_person,
			date_assigned = i_running_time,
	        dt_modified = i_running_time
		where (i_pk = otc_rc_id);
		
		let o_result = 1;
	
	elif i_origin_table = 'rms_cc_case' then
		select limit 1 role_name into i_assign_to_t from rms_param where param_grp_nm = 'cc-case' 
			and param_cd in (select task_status from rms_cc_case where cc_case_id = i_pk) and status = 'A';
		
		update rms_cc_case set
			assign_to = i_assign_to_t,
			pick_up = i_pickup_person,
	        modified_by = i_pickup_person,
			dt_pickup = i_running_time,
	    	dt_assigned = i_running_time,
	        dt_modified = i_running_time
		where cc_case_id = i_pk;
	
		execute procedure sp_inscccasehist(i_pk, i_running_time, i_pickup_person, 'Task pick up', 'PU') into o_result;
		if o_result < 1 then
			rollback work;
		elif o_result is null then
			rollback work;
			let o_result = -5;
		end if;
	end if;
	commit work;
	return o_result;
end procedure;

                        and t2.param_grp_nm = 'Billing-Status'
						and t1.dt_created is not null
						and (i_task_id is null or (t1.billing_no) like '%' || i_task_id || '%')
						and (i_task_desc is null or t1.billing_desc like '%' || i_task_desc || '%')
						and (i_requested_by is null or t3u2.nm like '%' || i_requested_by || '%')
						and (i_dt_requested is null or i_dt_requested = date(t1.dt_modified))
						and (i_status is null or t1.bil_wf_status = i_status)
                        and t2.role_name in s_userrole --like '%' || i_userrole ||'%' 
        );

        let rtt_total = (select count(1) from rms_rtt_wf t1 left outer join rms_param t2 on t1.rtt_status = t2.param_cd
    					left outer join rms_user tu1 on tu1.ssm4uuserrefno = t1.requested_by
    					left outer join rms_param tp1 on t1.refund_ty = tp1.param_cd
                        where (t1.status = 'A')
                        and t2.param_grp_nm = 'RTT-Status'
                        and t1.dt_requested is not null
                        and (i_task_id is null or (t1.rtt_app_no) like '%' || i_task_id || '%')
                        and (i_task_desc is null or tp1.nm_en like '%' || i_task_desc || '%')
                        and (i_requested_by is null or tu1.nm like '%' || i_requested_by || '%')
                        and (i_dt_requested is null or i_dt_requested = date(t1.dt_requested))
                        and (i_status is null or t2.param_cd = i_status)
                        and t2.role_name in s_userrole --like '%' || i_userrole ||'%' 
                        --and t1.assign_to in s_realroles
                        AND EXISTS (SELECT 1 FROM tt_realroles r WHERE (',' || t1.assign_to || ',') LIKE '%,' || r.role_nm_en || ',%')
                    );

        let otcrc_total = (select count(1) from rms_otc_rc t1 left outer join rms_param t2 on t1.rc_status = t2.param_cd and t2.param_grp_nm = 'OTC-RC'
    					left join rms_otc_counter t3 on t3.otc_counter_id = t1.otc_counter_id
    					left outer join rms_user tu1 on tu1.ssm4uuserrefno = t1.requested_by
                        where (t1.status = 'A')
                        and t2.param_grp_nm = 'OTC-RC'
	            		and t1.rc_status like '%PS%'
                        and t1.date_requested is not null
                        and (i_task_id is null or (t1.task_id) like '%' || i_task_id || '%')
                        and (i_task_desc is null or t2.nm_en like '%' || i_task_desc || '%')
                        and (i_requested_by is null or tu1.nm like '%' || i_requested_by || '%')
                        and (i_dt_requested is null or i_dt_requested = date(t1.date_requested))
                        and (i_status is null or t2.param_cd = i_status)
                        and t2.role_name in s_userrole --like '%' || i_userrole ||'%'
                        and t3.branch_cd in s_user_branch_cd
                    );
                   
		let ccc_total = (select count(1) from rms_cc_case t10 left outer join rms_param t11 on t10.task_status = t11.param_cd and t11.param_grp_nm = 'cc-case'
						left outer join rms_cc_cust t12 on t10.cc_cust_id = t12.cc_cust_id
						left join (select cc_case_id, modified_by, row_number() over (partition by cc_case_id order by dt_created desc) as rn,
			           	count(*) over (partition by cc_case_id) as tc from rms_cc_case_a) t13 on t10.cc_case_id = t13.cc_case_id and (t13.rn = 2 or t13.tc = 1)
						inner join rms_user t10u2 ON t10u2.ssm4uuserrefno = t13.modified_by
			            where t10.status = 'A'
                        and t11.param_grp_nm = 'cc-case'
			            and (t10.task_status in ('IP', 'PFSM', 'PFH', 'PSSS', 'PL', 'PCO', 'O', 'PD'))
			            and (i_task_id is null or (t10.task_no) like '%' || i_task_id || '%')
						and (i_task_desc is null or t11.nm_en like '%' || i_task_desc || '%')
						and (i_requested_by is null or t10u2.nm like '%' || i_requested_by || '%')
						and (i_dt_requested is null or i_dt_requested = date(t10.dt_modified))
						and (i_status is null or t10.task_status = i_status)
						and t11.role_name in s_userrole --like '%' || i_userrole ||'%'
                        AND EXISTS (SELECT 1 FROM tt_realroles r WHERE (',' || t10.assign_to || ',') LIKE '%,' || r.role_nm_en || ',%')
					);

        let total = bil_wf_total + bil_total + rtt_total + otcrc_total + ccc_total;
	
		foreach c_getpublictasks_n for
		
			select skip calculatedpage limit i_size task_desc, requested_by, dt_requested, pickup_by, dt_pick, task_status, origin_table, pk, task_key, user_role
				into o_task_desc, o_requested_by, o_dt_requested, o_pickup_by, o_dt_pick, o_task_status, o_origin_table, o_pk, o_task_key, userrole_to_check
			from (
				select t1.billing_desc as task_desc, t1u3.nm as requested_by, t1.dt_modified as dt_requested, t1u2.nm as pickup_by, t1.dt_pick as dt_pick, 
					(t2.nm_en || case when t1a.qr >= t1a.qfa then ' Query' else '' end) as task_status, 
					'rms_bil_wf' as origin_table, t1.bil_wf_id as pk, t1.billing_no as task_key, t2.role_name as user_role
				from rms_bil_wf t1 left outer join rms_param t2 on t1.bil_wf_status = t2.param_cd and t2.param_grp_nm = 'Billing-Status'
				left join (select bil_wf_id, max(case when msg_type = 'QR' then qc end) as QR, max(case when msg_type = 'QFA' then qc end) as QFA from   
					(select h.bil_wf_id, m.msg_type, count(m.msg_type) as qc from rms_bil_wf_hist h left join rms_bil_msg m on m.bil_wf_hist_id = h.bil_wf_hist_id
						where m.msg_type in ('QR', 'QFA') and h.status = 'A' group by h.bil_wf_id, m.msg_type) group by bil_wf_id) t1a on t1a.bil_wf_id = t1.bil_wf_id
				--left join rms_user t1u1 on t1u1.ssm4uuserrefno = t1.req_name
				left join rms_user t1u2 on t1u2.ssm4uuserrefno = t1.pickup_by
				left join (select bil_wf_id, modified_by, row_number() over (partition by bil_wf_id order by dt_created desc) as rn,
				           count(*) over (partition by bil_wf_id) as tc from rms_bil_wf_hist) t1b on t1.bil_wf_id = t1b.bil_wf_id and (t1b.rn = 2 or t1b.tc = 1)
				inner join rms_user t1u3 ON t1u3.ssm4uuserrefno = t1b.modified_by				
				where t1.status = 'A'
	            and t1.bil_wf_status = 'WF-N'
				and t1.dt_created is not null
				and (i_task_id is null or (t1.billing_no) like '%' || i_task_id || '%')
				and (i_task_desc is null or t1.billing_desc like '%' || i_task_desc || '%')
				and (i_requested_by is null or t1u3.nm like '%' || i_requested_by || '%')
				and (i_dt_requested is null or i_dt_requested = date(t1.dt_created))
				and (i_status is null or t1.bil_wf_status = i_status)
				and t2.role_name in s_userrole
				
				union 
	
	            select t3.billing_desc as task_desc, t3u2.nm as requested_by, t3.dt_modified as dt_requested, t3u1.nm as pickup_by, t3.dt_pick as dt_pick, 
    				(t4.nm_en || case when t3a.qcr > t3a.rcfa or t3a.qar > t3a.qafa then ' Query' else '' end) as task_status, 
	            	'rms_bil' as origin_table, t3.bil_id as pk, t3.billing_no as task_key, t4.role_name as user_role
				from rms_bil t3 left outer join rms_param t4 on t3.bil_wf_status = t4.param_cd and t4.param_grp_nm = 'Billing-Status'
				left join (select bil_id, max(case when msg_type = 'QCR' then qc end) as QCR, max(case when msg_type = 'RCFA' then qc end) as RCFA , 
					max(case when msg_type = 'QAR' then qc end) as QAR, max(case when msg_type = 'QAFA' then qc end) as QAFA from   
						(select h.bil_id, m.msg_type, count(m.msg_type) as qc from rms_bil_wf_hist h left join rms_bil_msg m on m.bil_wf_hist_id = h.bil_wf_hist_id
							where m.msg_type in ('QCR', 'QAR', 'RCFA', 'QAFA') and h.status = 'A' group by h.bil_id, m.msg_type) 
					group by bil_id) t3a on t3a.bil_id = t3.bil_id
				left join rms_user t3u1 on t3u1.ssm4uuserrefno = t3.pickup_by
				left join (select bil_id, modified_by, row_number() over (partition by bil_id order by dt_created desc) as rn,
				           count(*) over (partition by bil_id) as tc from rms_bil_wf_hist) t3b on t3.bil_id = t3b.bil_id and (t3b.rn = 2 or t3b.tc = 1)
				inner join rms_user t3u2 ON t3u2.ssm4uuserrefno = t3b.modified_by				
				where t3.status = 'A'
	            and (t3.bil_wf_status = 'WF-CN' or t3.bil_wf_status = 'WF-AN')
				and t3.dt_created is not null
				and (i_task_id is null or (t3.billing_no) like '%' || i_task_id || '%')
				and (i_task_desc is null or t3.billing_desc like '%' || i_task_desc || '%')
				and (i_requested_by is null or t3u2.nm like '%' || i_requested_by || '%')
				and (i_dt_requested is null or i_dt_requested = date(t3.dt_created))
				and (i_status is null or t3.bil_wf_status = i_status)
				and t4.role_name in s_userrole
				
				union 
				
				select t7.nm_en as task_desc, t5u1.nm as requested_by, t5.dt_requested as dt_requested, t5u2.nm as pickup_by, t5.date_pick as dt_pick, t6.nm_en as task_status, 'rms_rtt_wf' as origin_table, t5.rtt_wf_id as pk, t5.rtt_app_no as task_key, t6.role_name as user_role
					from rms_rtt_wf t5 left outer join rms_param t6 on t5.rtt_status = t6.param_cd left outer join rms_param t7 on t5.refund_ty = t7.param_cd
				left join rms_user t5u1 on t5u1.ssm4uuserrefno = t5.requested_by
				left join rms_user t5u2 on t5u2.ssm4uuserrefno = t5.pickup_by
				where (t5.status = 'A')
                and t6.param_grp_nm = 'RTT-Status'
	            and (t5.rtt_status like '%P%' or t5.rtt_status like '%T1%' or t5.rtt_status like '%T2%' or t5.rtt_status like '%T3%')
				and t5.dt_requested is not null
				and (i_task_id is null or (t5.rtt_app_no) like '%' || i_task_id || '%')
				and (i_task_desc is null or t7.nm_en like '%' || i_task_desc || '%')
				and (i_requested_by is null or t5u1.nm like '%' || i_requested_by || '%')
				and (i_dt_requested is null or i_dt_requested = date(t5.dt_requested))
				and (i_status is null or t6.param_cd = i_status)
				and t6.role_name in s_userrole
                --and t5.assign_to in s_realroles
	            AND EXISTS (SELECT 1 FROM tt_realroles r WHERE (',' || t5.assign_to || ',') LIKE '%,' || r.role_nm_en || ',%')

	            union 
	
	            --select t9.nm_en as task_desc, t8.requested_by as requested_by, t8.date_requested as dt_requested, t8.assigned_to as pickup_by, t8.date_assigned as dt_pick, t9.nm_en as task_status, 'rms_otc_rc' as origin_table, t8.otc_rc_id as pk, t8.task_id as task_key, t9.role_name as user_role
	            select t32.nm_en as task_desc, t30.nm as requested_by, t8.date_requested as dt_requested, t31.nm as pickup_by, t8.date_assigned as dt_pick, t9.nm_en as task_status, 'rms_otc_rc' as origin_table, t8.otc_rc_id as pk, t8.task_id as task_key, t9.role_name as user_role
	            	from rms_otc_rc t8 left outer join rms_param t9 on t8.rc_status = t9.param_cd and t9.param_grp_nm = 'OTC-RC'
	            left join rms_otc_counter t8a on t8a.otc_counter_id = t8.otc_counter_id
                left join rms_user t30 on t8.requested_by = t30.ssm4uuserrefno
                left join rms_user t31 on t8.assigned_to = t31.ssm4uuserrefno
			    left join rms_param t32 on t32.param_grp_nm = 'OTC-RC-Type' and t32.param_cd = 
			    	(case when t8.rc_type > 1 then (case when t8.rc_status == 'PS' then 2 
			    		when  t8.rc_status == 'PEMV' or t8.rc_status == 'PPC' or t8.rc_status == 'PCR' then 3 
			    		else t8.rc_type end) else t8.rc_type end)
	            where t8.status = 'A'
	            and t8.rc_status like '%PS%'
	            and t8.date_requested is not null
	            and (i_task_id is null or (t8.task_id) like '%' || i_task_id || '%')
				and (i_task_desc is null or t9.nm_en like '%' || i_task_desc || '%')
				and (i_requested_by is null or t30.nm like '%' || i_requested_by || '%')
				and (i_dt_requested is null or i_dt_requested = date(t8.date_requested))
				and (i_status is null or t9.param_cd = i_status)
                and t8a.branch_cd in s_user_branch_cd
				and t9.role_name in s_userrole
				
				union
				
				select t10.invoice_desc as task_desc, t10u2.nm as requested_by, t10.dt_modified as dt_requested, 
				       t10u1.nm as pickup_by, t10.dt_pickup as dt_pick, t11.nm_en as task_status, 'rms_cc_case' as origin_table, 
				       t10.cc_case_id as pk, t10.task_no as task_key, t11.role_name as user_role
				from rms_cc_case t10 
				left outer join rms_param t11 on t10.task_status = t11.param_cd and t11.param_grp_nm = 'cc-case' 
				left outer join rms_cc_cust t12 on t10.cc_cust_id = t12.cc_cust_id
				left join rms_user t10u1 on t10u1.ssm4uuserrefno = t10.pick_up
				left join (select cc_case_id, modified_by, row_number() over (partition by cc_case_id order by dt_created desc) as rn,
				           count(*) over (partition by cc_case_id) as tc from rms_cc_case_a) t13 on t10.cc_case_id = t13.cc_case_id and (t13.rn = 2 or t13.tc = 1)
				inner join rms_user t10u2 ON t10u2.ssm4uuserrefno = t13.modified_by
				where t10.status = 'A'
				and (t10.task_status in ('IP', 'PFSM', 'PFH', 'PSSS', 'PL', 'PCO', 'O', 'PD'))
				and t11.param_grp_nm = 'cc-case'
				and (i_task_id is null or (t10.task_no) like '%' || i_task_id || '%')
				and (i_task_desc is null or t11.nm_en like '%' || i_task_desc || '%')
				and (i_requested_by is null or t10u2.nm like '%' || i_requested_by || '%')
				and (i_dt_requested is null or i_dt_requested = date(t10.dt_created))
				and (i_status is null or t10.task_status = i_status)
				and t11.role_name in s_userrole
                AND EXISTS (SELECT 1 FROM tt_realroles r WHERE (',' || t10.assign_to || ',') LIKE '%,' || r.role_nm_en || ',%')
		)
	        order by dt_requested desc
			return o_task_key, o_task_desc, o_requested_by, o_dt_requested, o_pickup_by, o_dt_pick, o_task_status, o_origin_table, o_pk, total with resume;
		
		end foreach
		drop table tt_realroles;
		
end procedure;
